 26 var ScreenTrigger = Class.create(Trigger,
 27     /** @lends ScreenTrigger.prototype */ {
 28 
 29     /**
 30      * Screen onload Triggers representation
 31      * @constructs
 32      * @extends Trigger
 33      */
 34     initialize: function ($super, /** Object */ to) {
 35         var from = {
 36             'instance': ScreenTrigger.INSTANCE_NAME,
 37             'name': ScreenTrigger.ONLOAD
 38         };
 39         $super(from, to);
 40     },
 41 
 42     /**
 43      * @override
 44      */
 45     getSourceId: function() {
 46         return this._from.instance;
 47     },
 48 
 49     /**
 50      * @override
 51      */
 52     getSourceInstance: function() {
 53         var fakeInstance = {
 54             getTitle: function() {
 55                 return ScreenTrigger.INSTANCE_NAME;
 56             },
 57             getId: function() {
 58                 return ScreenTrigger.INSTANCE_NAME;
 59             }
 60         };
 61         return fakeInstance;
 62     },
 63 
 64 
 65     /**
 66      * Returns the JSON object representing the Trigger
 67      * @type Object
 68      * @override
 69      */
 70     toJSON: function() {
 71         return {
 72             'from': {
 73                 'buildingblock': "",
 74                 'name': "_onload"
 75             },
 76             'to': {
 77                 'buildingblock': this.getDestinationId(),
 78                 'action': this.getDestinationAction()
 79             }
 80         };
 81     },
 82 
 83     // **************** PRIVATE METHODS **************** //
 84 
 85     /**
 86      * Creates the trigger id
 87      * @private
 88      * @override
 89      * @type String
 90      */
 91     _createId: function (/** Object */ from, /** Object */ to) {
 92         return from.instance + from.name + to.instance.getId() + to.action;
 93     }
 94 });
 95 
 96 // Class attributes
 97 ScreenTrigger.INSTANCE_NAME = "Screen";
 98 ScreenTrigger.ONLOAD = "onload";
