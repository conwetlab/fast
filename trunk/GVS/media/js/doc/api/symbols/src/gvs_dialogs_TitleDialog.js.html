 26 var TitleDialog = Class.create(ConfirmDialog /** @lends ParamsDialog.prototype */, {
 27     /**
 28      * This class handles the dialog
 29      * to assign triggers to actions
 30      * @constructs
 31      * @extends ConfirmDialog
 32      */
 33     initialize: function($super, /** String */ title,
 34                             /** Function */ onChangeCallback) {
 35 
 36         $super("Change Screen title");
 37 
 38         this._title = title;
 39 
 40 
 41         this._onChangeCallback = onChangeCallback;
 42 
 43     },
 44 
 45     // **************** PUBLIC METHODS **************** //
 46 
 47     /**
 48      * This function returns the DOMNode of the button that shows the
 49      * dialog
 50      * @type DOMNode
 51      */
 52     getButtonNode: function() {
 53 
 54     	var button = new dijit.form.Button({
 55             'label': '...',
 56             'onClick': this.show.bind(this)
 57         });
 58 
 59         return new Element('div', {
 60             'class': 'triggerButton'
 61         }).update(button.domNode);
 62     },
 63 
 64     // **************** PRIVATE METHODS **************** //
 65 
 66 
 67     /**
 68      * initDialogInterface
 69      * This function creates the dom structure
 70      * @private
 71      * @override
 72      */
 73     _initDialogInterface: function () {
 74         this._setHeader(
 75                 "Change Screen title",
 76                 "This title will be shown when executing the gadget"
 77         );
 78 
 79         var formData = [
 80             {
 81                 'type':'input',
 82                 'label': 'Screen Title:',
 83                 'name': 'title',
 84                 'value': this._title,
 85                 'message': 'Title cannot be blank',
 86                 'required': true
 87             }
 88         ];
 89 
 90         this._setContent(formData);
 91     },
 92 
 93     /**
 94      * Overriding onOk handler
 95      * @private
 96      * @override
 97      */
 98     _onOk: function($super) {
 99     	$super();
100         this._onChangeCallback(this._getForm().title.value);
101     }
102 });
