1 var Pipe = Class.create(
  2     /** @lends Pipe.prototype */ {
  3 
  4     /**
  5      * Pipes graphical representation
  6      * @constructs
  7      */ 
  8     initialize: function (/** WireIt.Wire */ wire, /** String */ id) {
  9 
 10         /**
 11          * Pipe wire
 12          * @private
 13          * @type WireIt.Wire
 14          */
 15         this._wire = wire;
 16 
 17         /**
 18         * Pipe id
 19         * @private
 20         * @type String
 21         */
 22         this._id = id;
 23 
 24         /**
 25          * Valid pipe
 26          * @type Boolean
 27          * @private
 28          */
 29         this._isValid = true;
 30     },
 31 
 32     /**
 33      * Returns the WireIt wire object
 34      * @type WireIt.Wire
 35      */
 36     getWire: function() {
 37         return this._wire;
 38     },
 39 
 40     /**
 41      * Returns the pipe id
 42      * @type String
 43      */
 44     getId: function() {
 45         return this._id;
 46     },
 47 
 48     /**
 49      * Returns the source terminal of the pipe
 50      * @type Terminal
 51      */
 52     getSource: function() {
 53         return this._wire.terminal1;
 54     },
 55 
 56 
 57     /**
 58      * Returns the destination terminal of the pipe
 59      * @type Terminal
 60      */
 61     getDestination: function() {
 62         return this._wire.terminal2;
 63     },
 64 
 65     /**
 66      * Gets the JSON object representing the pipe for screen definition
 67      * @type Object
 68      */
 69     getJSONforScreen: function() {
 70         return {'from': {
 71                     'buildingblock': this.getSource().getBuildingBlockId(),
 72                     'condition': this.getSource().getConditionId()
 73                 },
 74                 'to': {
 75                     'buildingblock': this.getDestination().getBuildingBlockId(),
 76                     'action': this.getDestination().getActionId(),
 77                     'condition': this.getDestination().getConditionId()
 78                 }
 79             };
 80     },
 81 
 82     /**
 83      * Gets the JSON object representing the pipe for catalogue check
 84      * @type Object
 85      */
 86     getJSONforCheck: function() {
 87         return {
 88                 'from': {
 89                     'buildingblock': this.getSource().getBuildingBlockUri(),
 90                     'condition': this.getSource().getConditionId()
 91                 },
 92                 'to': {
 93                     'buildingblock': this.getDestination().getBuildingBlockUri(),
 94                     'action': this.getDestination().getActionId(),
 95                     'condition': this.getDestination().getConditionId()
 96                 }
 97             };
 98     },
 99 
100     /**
101      * Sets the reachability coloring information of the pipe
102      */
103     setReachability: function(/** Object */ reachabilityData) {
104         var options = this._wire.options;
105         var satisfied = reachabilityData.satisfied;
106         if (!reachabilityData.correct) {
107             options = Object.extend(options, {
108                 'color': '#EAEAEA',
109                 'bordercolor': '#808080'
110             });
111         } else {
112             if (satisfied) {
113                 options = Object.extend(options, {
114                     'color': '#DDF7DD',
115                     'bordercolor': '#008000'
116                 });
117             } else {
118                 options = Object.extend(options, {
119                     'color': '#F5D9D9',
120                     'bordercolor': '#B90000'
121                 });
122             }
123         }
124         this._wire.setOptions(options);
125         this._wire.redraw();
126 
127         if (!reachabilityData.correct) {
128              this._wire.element.setStyle({'opacity': '0.5'});
129              if (this._isValid) {
130                  Utils.showMessage("The elements you have connected are not compatible." +
131                             " It may not work", {
132                             'hide': true,
133                             'error': true
134                             });
135                  this._isValid = false;
136              }
137              
138         }
139     },
140 
141     /**
142      * Is the pipe semantically correct
143      * @type Boolean
144      */
145     isValid: function() {
146         // TODO: check if there are more conditions
147         return (this.getSource().getBuildingBlockId() != this.getDestination().getBuildingBlockId())
148                 && this._isValid;
149     },
150 
151 
152     /**
153      * Destroys the wire
154      */
155     destroy: function() {
156         this.getSource().removeWire(this._wire);
157         this.getDestination().removeWire(this._wire);
158     },
159 
160 
161     /**
162      * Sets the pipe visible or hidden. To be called whenever a document
163      * is selected or deselected
164      */
165     setVisible: function(/** Boolean */ visible) {
166         if (visible) {
167             this.getSource().updatePosition();
168             this.getDestination().updatePosition();
169         }
170         var style = {
171             'visibility': visible ? 'visible': 'hidden'
172         }
173         this._wire.element.setStyle(style);
174     }
175     
176     // **************** PRIVATE METHODS **************** //
177 
178 });
