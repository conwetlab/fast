1 var Table = Class.create( /** @lends Table.prototype */ {
  2     /**
  3      * This abstract class handles the different properties tables
  4      * belonging to different documents
  5      * @abstract
  6      * @constructs
  7      */ 
  8     initialize: function(/** DOMNode */ parentNode, /** String */ baseTitle, 
  9             /** String */ region, /** String (optional) */ minSize) {
 10         /** 
 11          * Parent node
 12          * @type DOMNode
 13          * @private @member
 14          */
 15         this._parentNode = parentNode;
 16         
 17         /** 
 18          * Title of the table
 19          * @type String
 20          * @private @member
 21          */
 22         this._title = baseTitle;
 23         
 24         /**
 25          * Node of the title
 26          * @type DOMNode
 27          * @private @member 
 28          */
 29         this._titleNode = new Element ('div',{
 30             'class': 'dijitAccordionTitle'
 31         }).update(this._title);
 32         
 33         /** 
 34          * Node of the table
 35          * @type DOMNode
 36          * @private @member
 37          */      
 38         this._tableNode = new Element ('table',{
 39             'class': 'properties_table'
 40         });
 41         
 42         var container= new dijit.layout.ContentPane({
 43             'region': region,
 44             'splitter': true
 45         });
 46         if (minSize) {
 47             container.attr('minSize', minSize);
 48         } 
 49         
 50         container.domNode.insert(this._titleNode);
 51         container.domNode.insert(this._tableNode);
 52         this._parentNode.addChild(container);
 53     },
 54     
 55 
 56     // **************** PUBLIC METHODS **************** //
 57 
 58     /**
 59      * Set the table title
 60      */
 61     setTitle: function (/** String */ title) {
 62         this._title = title;
 63         this._titleNode.update(title);
 64     },
 65 
 66     /**
 67      * Returns the table title
 68      * @type String
 69      */    
 70     getTitle: function () {
 71         return this._title;
 72     },
 73 
 74     /**
 75      * Returns the table Node
 76      * @type DOM
 77      */    
 78     getTableNode: function () {
 79         return this._tableNode;
 80     },
 81 
 82     /**
 83      * Sets the field titles
 84      * @type String
 85      */    
 86     insertFieldTitles: function (/** Array */ fieldTitles) {
 87         var tr = new Element('tr', {
 88             'class': 'tableHeader'
 89         });
 90         
 91         fieldTitles.each (function(title){
 92            var td = new Element ('td').update(title);
 93            tr.insert(td);
 94         });
 95         
 96         this._tableNode.insert(tr);
 97     },
 98 
 99     /**
100      * Updates the data in the table
101      */      
102     insertDataValues: function (/** Hash */ data)  {
103         
104         data.each (function(line){
105             var tr = new Element('tr');
106             
107             //TODO: What happen with the classes, maybe another method?
108             line.each (function(field){
109                var td = new Element ('td');
110                var div = new Element ('div').update(field);
111                if (typeof field === "string") {
112                    div.setAttribute('title', field);
113                }
114                td.insert(div)
115                tr.insert(td);
116             });
117             this._tableNode.insert(tr);
118         }.bind(this));
119     },
120 
121     /**
122      * Empty the data in the table
123      */      
124     emptyTable : function (){
125         this._tableNode.update("");
126     }
127     
128     
129 
130     // **************** PRIVATE METHODS **************** //
131 
132     
133 });
