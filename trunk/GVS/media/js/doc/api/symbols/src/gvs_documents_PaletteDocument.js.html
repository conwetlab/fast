1 var PaletteDocument = Class.create(AbstractDocument, /** @lends PaletteDocument.prototype */ {
  2     /**
  3      * Represents a document and its tab. Subclasses must provide the 
  4      * inner content.
  5      * @abstract
  6      * @extends AbstractDocument
  7      * @constructs
  8      */ 
  9     initialize: function($super,
 10             /** String */ title, 
 11             /** Array */ validBuildingBlocks,
 12             /** Array */ domainContext) {
 13         $super(title);
 14         
 15         /**
 16          * List of tags
 17          * @type Array
 18          */
 19         this._domainContext = domainContext;
 20         
 21         /**
 22          * Main border container
 23          * @type dijit.layout.BorderContainer
 24          * @private @member
 25          */
 26         this._mainBorderContainer = null;
 27         
 28         var buildingBlockSets = new Array();
 29         
 30         $A(validBuildingBlocks).each(function(blockType) {
 31             var buildingBlockSet;
 32             switch (blockType) {
 33                 case Constants.BuildingBlock.SCREEN:
 34                     buildingBlockSet = new ScreenSet(this._domainContext);
 35                     break;
 36                     
 37                 case Constants.BuildingBlock.DOMAIN_CONCEPT:
 38                     buildingBlockSet = new DomainConceptSet(this._domainContext);
 39                     break;
 40 
 41                 default:
 42                     throw "Unexpected type of building block: " + blockType +
 43                             ". PaletteDocument::initialize";
 44             }
 45             
 46             buildingBlockSets.push(buildingBlockSet);
 47         }.bind(this));
 48         
 49         this._renderMainUI();
 50         
 51         /**
 52          * InferenceEngine
 53          * @type InferenceEngine
 54          * @private @member
 55          */ 
 56         this._inferenceEngine = new InferenceEngine();
 57         
 58          /**
 59          * This property represents the selected element
 60          * @type BuildingBlockInstance
 61          * @private @member
 62          */
 63         this._selectedElement = null;
 64                       
 65         /**
 66          * Palette Controller
 67          * @type PaletteController
 68          * @private @member
 69          */ 
 70         this._paletteController = new PaletteController(buildingBlockSets,
 71                 /* (DropZone) */this, this._inferenceEngine);
 72         
 73         this._renderPaletteArea();
 74         
 75     },
 76     
 77     /**
 78      * Returns the selected element for the screenflow document
 79      * @type ComponentInstance
 80      */
 81     getSelectedElement: function () {
 82         return this._selectedElement;
 83     },
 84 
 85     /**
 86      * Select a screen in the screenflow document
 87      * @param ComponentInstance
 88      *      Element to be selected for the
 89      *      Screenflow document.
 90      */
 91     setSelectedElement: function (element) {
 92         if (this._selectedElement != null) {
 93             this._selectedElement.getView().setSelected(false);
 94         }
 95         
 96         if (element != undefined) {
 97             this._selectedElement = element;
 98             this._selectedElement.getView().setSelected(true);
 99         } else {
100             this._selectedElement = null;
101         }
102     },
103         
104     /**
105      * Implementing DropZone interface.
106      * To be overriden.
107      */
108     drop: function(/** Object */ droppedElement) {
109         throw "Abstract method invation. PaletteDocument::drop";
110     },
111     
112     /**
113      * Keypress event handler
114      */
115     onKeyPressed: function(/** Event */ e) {    
116         
117         switch(e.keyCode) {
118             case Event.KEY_DELETE:
119                 if (this._selectedElement != null) { //Delete an element from the canvas
120                     var title = null;
121                     if (this._selectedElement.getTitle()){
122                         title = 'the element "' + this._selectedElement.getTitle() + '"';
123                     } else {
124                         title = "the selected element";
125                     }
126                     
127                     confirm("You are about to remove " + title + " from canvas. Are you sure?", 
128                             function(/** Boolean */ confirmed) {
129                                 if (confirmed) {
130                                     this._deleteSelectedElement();
131                                 }
132                             }.bind(this)
133                     );
134                 }
135                 break; 
136         }
137         
138     },
139

161     },

163     /**
164      * Constructs the document content.
165      * @private
166      */
167     _renderMainUI: function(){
168 
169         this._mainBorderContainer = new dijit.layout.BorderContainer({
170             design:"sidebar",
171             liveSplitters:"false",
172             splitter:"true"
173         });
174        
175         this._mainBorderContainer.addChild(this._renderCenterContainer());
176 
177         this._tab.setContent(this._mainBorderContainer.domNode);
178     },
179     /**
180      * Renders the palette area
181      */
182     _renderPaletteArea: function() {
183         this._mainBorderContainer.addChild(this._paletteController.getNode());
184     },
185 
186     _deleteSelectedElement: function() {
187         if (this._selectedElement != null) {
188             this._deleteInstance(this._selectedElement);
189             this.setSelectedElement();
190         }
191     }    
192 });
