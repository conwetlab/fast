1 var ManageScreensDialog = Class.create(GalleryDialog /** @lends ManageScreensDialog.prototype */, {
  2     /**
  3      * This class handles the dialog
  4      * to open an existing screen
  5      * @constructs
  6      * @extends GalleryDialog
  7      */ 
  8     initialize: function($super) {  
  9         $super("Screen browser", {'onDblClick': this._openScreen.bind(this),
 10                                   'disableIfNotValid': true });
 11 
 12         /**
 13          * List of screens
 14          * @type Array
 15          * @private
 16          */
 17         this._screens = null;
 18     },
 19     
 20     
 21     // **************** PUBLIC METHODS **************** //
 22 
 23     /**
 24      * @override
 25      */
 26     show: function() {
 27         PersistenceEngine.sendGet(URIs.screen, this, this._onLoadSuccess, Utils.onAJAXError);
 28     },
 29 
 30 
 31     // **************** PRIVATE METHODS **************** //
 32 
 33 
 34     /** 
 35      * initDialogInterface
 36      * This function creates the dom structure
 37      * @private
 38      * @override
 39      */
 40     _initDialogInterface: function (){
 41 
 42         this._setHeader("Browse your screens",
 43                 "These are the screens you have created. Here you can " +
 44                 "continue editing them and share your work with the community");
 45 
 46         this._setFields([{
 47                 'title': 'Icon',
 48                 'className': 'icon'
 49             }, {
 50                 'title': 'Screen Name',
 51                 'className': 'name'
 52             }, {
 53                 'title': 'Screen Version',
 54                 'className': 'version'
 55             }, {
 56                 'title': 'Tags',
 57                 'className': 'tags'
 58             }, {
 59                 'title': 'Description',
 60                 'className': 'description'
 61             }, {
 62                 'title': 'Sharing',
 63                 'className': 'sharing'
 64             }
 65         ]);
 66 
 67         this._setButtons([{
 68                 'value': 'Open screen',
 69                 'handler': this._openScreen.bind(this),
 70                 'disableIfNotValid': true
 71             }, {
 72                 'value': 'Share/Unshare screen',
 73                 'handler': this._shareScreen.bind(this)
 74             }, {
 75                 'value': 'Delete screen',
 76                 'handler': this._deleteScreen.bind(this)
 77             }/*, {
 78                 'value': 'Add external screen',
 79                 'handler': this._addScreen.bind(this)
 80             }*/]);
 81 
 82         this._createScreenList();
 83         this._render();
 84     },
 85 
 86     /**
 87      * Creates the the screen list to be handled by its parent class
 88      * @private
 89      */
 90     _createScreenList: function() {
 91         this._emptyRows();
 92         this._screens.each(function(screen) {
 93             this._addRow({
 94                             'key': screen.id,
 95                             'values': [
 96                                 new Element('img', {'src': screen.icon}),
 97                                 	screen.name,
 98                                 '<span class="bold">Version: </span>' +
 99                                      screen.version,
100                                 '<span class="bold">Tags: </span>' +
101                                 	screen.tags.collect(function(tag) {
102                                         return tag.label['en-gb'];
103                                     }).join(", "),
104                                 '<span class="bold">Description </span><br />'+
105                                 	screen.description['en-gb'],
106                                 '<span class=' + (screen.uri ? '"shared"': '"unshared"') +
107                                     '> </span>'
108                              ],
109                              'isValid': screen.definition ? true : false
110                         });
111         }.bind(this));
112     },
113     
114     /**
115      * On Success handler
116      * @private
117      */
118     _onLoadSuccess: function(/** XMLHttpRequest */ transport) {
119         this._screens = JSON.parse(transport.responseText);
120         this._show();
121     },
122 
123     /**
124      * On Success handler, when reload
125      * @private
126      */
127     _onReLoadSuccess: function(/** XMLHttpRequest */ transport) {
128         this._screens = JSON.parse(transport.responseText);
129         this._createScreenList();
130         this._render(false);
131     },
132 
133     /**
134      * Open a screen by its id
135      * @private
136      */
137     _openScreen: function(/** String */ id) {
138         var documentController = GVS.getDocumentController();
139         documentController.loadScreen(id);
140         this._dialog.hide();
141     },
142 
143     /**
144      * Share or unshare a screen depending of its status
145      * @private
146      */
147     _shareScreen: function(/** String */ id) {
148         var uri = URIs.share.replace("<id>", id);
149 
150         var screen = this._screens.detect(function(element) {
151             return element.id == id;
152         });
153         if (screen.uri) {
154             // Unshare screen       
155             PersistenceEngine.sendDelete(uri, this, this._reload, Utils.onAJAXError);
156         } else {
157             // Share screen
158             PersistenceEngine.sendPost(uri, null, null, this, this._reload,
159                                         Utils.onAJAXError);
160         }
161     },
162 
163     /**
164      * Starts the deletion of a screen
165      * @private
166      */
167     _deleteScreen: function(/** String */ id) {
168         confirm("Are you sure you want to delete the screen? This action cannot " +
169         "be undone", this._confirmDelete.bind({'mine': this, 'id': id}));
170     },
171 
172     /**
173      * Execute the deletion of the screen
174      * @private
175      */
176     _confirmDelete: function() {
177         var uri = URIs.buildingblock + this.id;
178         PersistenceEngine.sendDelete(uri, this.mine, this.mine._reload, Utils.onAJAXError);
179     },
180 
181     /**
182      * Starts the process of adding an external screen
183      * @private
184      */
185     _addScreen: function() {
186         this._dialog.hide();
187         //GVS.action("addScreen");
188     },
189 
190     /**
191      * Reloads the screen list
192      * @private
193      */
194     _reload: function() {
195         PersistenceEngine.sendGet(URIs.screen, this, this._onReLoadSuccess, Utils.onAJAXError);
196     }
197 });
