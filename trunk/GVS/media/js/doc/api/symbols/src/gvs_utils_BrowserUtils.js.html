 26 var BrowserUtils = Class.create();
 27 
 28 // **************** STATIC METHODS **************** //
 29 
 30 Object.extend(BrowserUtils, {
 31 
 32     /**
 33      * Gets browser window height
 34      * @type Integer
 35      */
 36     getHeight : function () {
 37         var newHeight=window.innerHeight; //Non-IE (Firefox and Opera)
 38 
 39         if (document.documentElement &&
 40                 document.documentElement.clientHeight) {
 41           //IE 6+ in 'standards compliant mode'
 42           newHeight = document.documentElement.clientHeight;
 43 
 44         } else if( document.body && document.body.clientHeight ) {
 45           //IE 4 compatible and IE 5-7 'quirk mode'
 46           newHeight = document.body.clientHeight;
 47         }
 48 
 49         return newHeight;
 50     },
 51 
 52 
 53     /**
 54      * Gets browser window width
 55      * @type Integer
 56      */
 57     getWidth : function(){
 58         var newWidth=window.innerWidth; //Non-IE (Firefox and Opera)
 59 
 60         if (document.documentElement &&
 61                 (document.documentElement.clientWidth ||
 62                  document.documentElement.clientHeight)) {
 63             //IE 6+ in 'standards compliant mode'
 64             newWidth = document.documentElement.clientWidth;
 65         } else if (document.body &&
 66                 (document.body.clientWidth || document.body.clientHeight)) {
 67             //IE 4 compatible and IE 5-7 'quirk mode'
 68             newWidth = document.body.clientWidth;
 69         }
 70 
 71         return newWidth;
 72     },
 73 
 74 
 75     /**
 76      * Determines if a button code is the left one
 77      * (right button for left-handed people).
 78      * @type Boolean
 79      */
 80     isLeftButton : function(button) {
 81 
 82         if (button == 0 || (this.isIE() && button == 1)) {
 83             return true;
 84         } else {
 85             return false;
 86         }
 87     },
 88 
 89 
 90     /**
 91      * Is the browser Internet Explorer?
 92      * @type Boolean
 93      */
 94     isIE : function(){
 95         if (/MSIE (\d+\.\d+);/.test(navigator.userAgent)) { //test for MSIE x.x;
 96             return true;
 97         } else {
 98             return false;
 99         }
100     }
101 });
