1 var ScreenflowPlayer = Class.create( /** @lends ScreenflowPlayer.prototype */ {
  2     /**
  3      * On charge of playing screenflows
  4      * @constructs
  5      */
  6     initialize: function() {
  7 
  8         /**
  9          * @type ScreenflowDescription
 10          * @private @member
 11          */
 12         this._description = null;
 13 
 14         /**
 15          * @type PreviewDialog
 16          * @private @member
 17          */
 18         this._dialog = null;
 19 
 20         /**
 21          * Is the logging (debugging) enabled?
 22          * @private
 23          * @type Boolean
 24          */
 25         this._logEnabled = false;
 26 
 27         /**
 28          * DOM node of the player
 29          * @private
 30          * @type DOMNode
 31          */
 32         this._object = null;
 33 
 34     },
 35 
 36 
 37     // **************** PUBLIC METHODS **************** //
 38 
 39     /**
 40      * Shows the Screenflow execution
 41      * @public
 42      */
 43     playScreenflow: function (/** ScreenflowDescription */ description) {
 44         this._description = description;
 45 
 46         if (!this._dialog) {
 47             var title = this._description.name;
 48 
 49             this._dialog = new PreviewDialog(title, this._getPreview());
 50         } else {
 51             this._dialog.setContent(this._getPreview());
 52         }
 53         this._dialog.show();
 54 
 55     },
 56 
 57     /**
 58      * Debug screenflow in a new window
 59      */
 60     debugScreenflow: function(/** ScreenflowDescription*/ description) {
 61         this._description = description;
 62         GVS.getDocumentController().openExternalTool("Screenflow Debugger",
 63             this._getScreenflowURL("debug"));
 64     },
 65 
 66     // **************** PRIVATE METHODS **************** //
 67 
 68     /**
 69      * This method creates a DOM Node with the preview
 70      * of the Screenflow
 71      * @type DOMNode
 72      */
 73     _getPreview: function() {
 74         var node = new Element('div', {
 75             'class': 'player'
 76         });
 77 
 78         var errorField = new Element('div', {
 79             'class': 'error'
 80         });
 81         node.appendChild(errorField);
 82 
 83         this._object = new Element ('object', {
 84             'data': this._getScreenflowURL(),
 85             'class': 'embed'
 86         });
 87 
 88         node.appendChild(this._object);
 89 
 90         var bottomZone = new Element('div');
 91         var linkNode = new Element("a", {
 92             "href": this._getScreenflowURL("debug"),
 93             "target": "_blank"
 94         }).update("[Debug in new window]");
 95         bottomZone.appendChild(linkNode);
 96         bottomZone.appendChild(new Element("br"));
 97 
 98         var loggingCheckBox = new dijit.form.CheckBox({
 99             checked: this._logEnabled
100         });
101         bottomZone.appendChild(loggingCheckBox.domNode);
102         loggingCheckBox.domNode.observe("change",
103                 this._toggleLogging.bind(this));
104 
105         var label = new Element('span')
106                         .update("Logging enabled (better if you have Firebug)");
107         bottomZone.appendChild(label);
108 
109         node.appendChild(bottomZone);
110         return node;
111     },
112     /**
113      * Toggle the logging
114      * @private
115      */
116     _toggleLogging: function(/** Event*/ e) {
117         var checkbox = Event.element(e);
118         this._logEnabled = !this._logEnabled;
119         checkbox.checked = this._logEnabled;
120         this._object.contentDocument.location.href = this._getScreenflowURL();
121     },
122 
123     /**
124      * Gets the screenflow URL
125      * @type String
126      * @private
127      */
128     _getScreenflowURL: function(_debugLevel) {
129         var debugLevel = _debugLevel || (this._logEnabled  ? "logging" : "");
130         return URIs.storePlayScreenflow + "?screenflow=" +
131             this._description.getId() + "&debugLevel=" + debugLevel;
132     }
133 });
