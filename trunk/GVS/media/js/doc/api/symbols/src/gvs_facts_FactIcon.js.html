 26 var FactIcon = Class.create( /** @lends FactIcon.prototype */ {
 27 
 28     /**
 29      * Graphical representation of a fact.
 30      * @constructs
 31      * @param Fact fact
 32      * @param String size  Icon size ("inline"|"embedded"|"standalone")
 33      */
 34     initialize: function(fact, size) {
 35         /**
 36          * Fact
 37          * @type Fact
 38          * @private
 39          */
 40         this._fact = fact;
 41 
 42         /**
 43          * Icon size
 44          * @type String
 45          * @private
 46          */
 47         this._size = size;
 48 
 49         /**
 50          * Fact icon root node
 51          * @type DOMNode
 52          * @private
 53          */
 54         this._node = new Element ("div", {
 55                 "class": this._size + "_fact fact"
 56                 });
 57 
 58         var contentLayer = new Element("div", {
 59                 "class": "contentLayer"
 60                 });
 61         this._node.appendChild(contentLayer);
 62         contentLayer.update(this._fact.getShortcut());
 63 
 64         var recommendationLayer = new Element("div", {
 65             "class": "recommendationLayer"
 66             });
 67         this._node.appendChild(recommendationLayer);
 68     },
 69 
 70 
 71     // **************** PUBLIC METHODS **************** //
 72 
 73 
 74     /**
 75      * Gets the root node.
 76      * @type DOMNode
 77      * @public
 78      */
 79     getNode: function () {
 80         return this._node;
 81     },
 82      /**
 83      * Gets the fact object
 84      * @type Fact
 85      * @public
 86      */
 87     getFact: function () {
 88         return this._fact;
 89     }
 90 
 91     // **************** PRIVATE METHODS **************** //
 92 });
