 26 var ScreenFactory = Class.create(BuildingBlockFactory,
 27     /** @lends ScreenFactory.prototype */ {
 28 
 29     /**
 30      * Factory of screen building blocks.
 31      * @constructs
 32      * @extends BuildingBlockFactory
 33      */
 34     initialize: function($super) {
 35         $super();
 36 
 37          /**
 38          * Hash table (organized by URI)
 39          * containing all the BB descriptions
 40          * @type Hash
 41          * @private @member
 42          */
 43         this._buildingBlockDescriptions = new Hash();
 44 
 45     },
 46 
 47     // **************** PUBLIC METHODS **************** //
 48     /**
 49      * Gets the type of building block this factory mades.
 50      * @type String
 51      * @override
 52      */
 53     getBuildingBlockType: function (){
 54         return Constants.BuildingBlock.SCREEN;
 55     },
 56 
 57 
 58     /**
 59      * Gets building block descriptions by URI
 60      * @type {BuildingBlockDescription[]}
 61      * @override
 62      */
 63     getBuildingBlocks: function (/** Array */ uris, _includeCopies) {
 64         var includeCopies = Utils.variableOrDefault(_includeCopies, true);
 65         var result = new Array();
 66         var validUris = $A(uris);
 67         if (!includeCopies) {
 68             validUris = validUris.findAll(function(uri) {
 69                 return (uri.indexOf("clones") == -1);
 70             });
 71         }
 72         validUris.each(function(uri){
 73             if(this._buildingBlockDescriptions.get(uri)) {
 74                 result.push(this._buildingBlockDescriptions.get(uri));
 75             } else {
 76                 throw "Ooops. Something went wrong. " +
 77                       "ScreenFactory::getBuildingBlocks";
 78             }
 79         }.bind(this));
 80         return result;
 81     },
 82 
 83     /**
 84      * This function retrieves the pending elements from the serverside
 85      * catalogue
 86      */
 87     cacheBuildingBlocks: function (/** Array */ uris, /** Function */ callback){
 88         //URIs not already retrieved
 89         var pendingURIs = new Array();
 90         $A(uris).each (function (uri){
 91             if (!this._buildingBlockDescriptions.get(uri)){
 92                 pendingURIs.push (uri);
 93             }
 94         }.bind(this));
 95 
 96         if (pendingURIs.size() > 0) {
 97             var postData = Object.toJSON(pendingURIs);
 98             PersistenceEngine.sendPost(URIs.catalogueGetMetadata,
 99                 null, postData,
100                 {
101                     'mine': this,
102                     'callback': callback
103                 },
104                 this._onSuccess, Utils.onAJAXError);
105         } else {
106             callback();
107         }
108     },
109 
110     /**
111      * Gets a new instance of the type of the factory
112      * @override
113      * @type ScreenInstance
114      */
115     getInstance: function(/** BuildingBlockDescription */description, /** InferenceEngine */ engine) {
116         return new ScreenInstance(description, engine);
117     },
118 
119     // **************** PRIVATE METHODS **************** //
120 
121 
122 
123     /**
124      * Callback function
125      */
126     _onSuccess: function(/**XMLHttpRequest*/ transport) {
127         var metadata = transport.responseText.evalJSON();
128         //update the Screen Factory
129         this.mine._updateBuildingBlockDescriptions(metadata.screens);
130         //call the callback function passed as argument
131         this.callback();
132     },
133 
134     /**
135      * This function creates the different screen Descriptions
136      * @private
137      */
138     _updateBuildingBlockDescriptions: function (/** Array */ screenDescriptions) {
139 
140         for (var i=0; i< screenDescriptions.length ; i++) {
141             this._buildingBlockDescriptions.set(screenDescriptions[i].uri,
142                                         new ScreenDescription (screenDescriptions[i]));
143         }
144     }
145 });
