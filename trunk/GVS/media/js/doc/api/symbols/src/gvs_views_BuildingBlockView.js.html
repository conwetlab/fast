 26 var BuildingBlockView = Class.create( /** @lends BuildingBlockView.prototype */ {
 27     /**
 28      * This interface is met by all the building block graphical representations.
 29      * @abstract
 30      * @constructs
 31      */
 32     initialize: function() {
 33         /**
 34          * DOM Node
 35          * @type DOMNode
 36          * @private @member
 37          */
 38         this._node = null;
 39 
 40     },
 41 
 42     // **************** PUBLIC METHODS **************** //
 43 
 44 
 45     /**
 46      * getNode
 47      * @type DOMNode
 48      * @public
 49      */
 50     getNode: function () {
 51         return this._node;
 52     },
 53 
 54     /**
 55      * Sets the title if possible
 56      */
 57     setTitle: function(title) {
 58         // Do nothing
 59     },
 60 
 61     /*
 62      * return the dom element for use in the prototype.js functions
 63      * @type DOMNode
 64      * @public
 65      */
 66     toElement: function() {
 67         return this._node;
 68     },
 69 
 70     /**
 71      * Colorize the component depending on the reachability
 72      * @public
 73      */
 74     setReachability: function( /** Hash */ reachabilityData) {
 75         throw 'Abstract Method invocation. ' +
 76             'BuildingBlockView :: setReachability';
 77     },
 78 
 79     setSelected: function(/** Boolean */ selected) {
 80         if (selected) {
 81             this._node.addClassName('selected');
 82         } else {
 83             this._node.removeClassName('selected');
 84         }
 85     },
 86 
 87     /**
 88      * Removes the DOM Elements and frees building blocks
 89      */
 90     destroy: function () {
 91         this.remove();
 92         this._node = null;
 93     },
 94 
 95     /**
 96      * Removes the DOM Elements off the DOM Document
 97      */
 98     remove:function() {
 99         try {
100             this._node.remove();
101         } catch (e) {
102             // Do nothing
103         }
104     },
105 
106     /**
107      * Adds an event listener
108      */
109     addEventListener: function (/** Function */ handler, /** String */ event){
110         Element.observe(this._node, event, handler);
111     },
112 
113     /**
114      * Adds a new div layer on top of the view, covering it, in order
115      * to handle onclick and onmousedown events, when
116      * the view includes some terminals, as they consume
117      * these events by default
118      */
119     addGhost: function() {
120         var ghost = new Element('div', {
121             'class': 'ghost ghostLayer'
122         });
123         this._node.appendChild(ghost);
124     },
125 
126     /**
127      * set position
128      */
129     setPosition: function(position) {
130         var top = position.top;
131         var left = position.left;
132         this._node.setStyle({
133             'position': 'absolute',
134             'top': top + 'px',
135             'left': left + 'px'
136         });
137         return this;
138     },
139 
140     // **************** PRIVATE METHODS **************** //
141 
142     _setViewReachability: function(/** Object */ reachability,
143                             /** Hash */ preHash, /** Array */ postList,
144                             /** DOMNode */ node) {
145 
146         var satisfeable = reachability.reachability;
147         Utils.setSatisfeabilityClass(node, satisfeable);
148 
149         postList.each(function(post) {
150             Utils.setSatisfeabilityClass(post.getNode(), satisfeable);
151         });
152 
153         reachability.actions.each(function(action) {
154             action.preconditions.each(function(pre) {
155                 var preNode = preHash.get(pre.id).getNode();
156                 Utils.setSatisfeabilityClass(preNode, pre.satisfied);
157             }.bind(this));
158         }.bind(this));
159     }
160 });
