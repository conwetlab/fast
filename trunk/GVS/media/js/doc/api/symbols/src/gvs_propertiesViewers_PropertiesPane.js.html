1 var PropertiesPane = Class.create( /** @lends PropertiesPane.prototype */ {
  2     /**
  3      * This class handles the properties pane
  4      * @constructs
  5      */ 
  6     initialize: function(/** DOMNode */ parentNode) {
  7         /** 
  8          * Variable
  9          * @type Table
 10          * @private @member
 11          */
 12         this._propertiesTable = new Table(parentNode, 'Properties', 'left');
 13         
 14         this._propertiesTable.insertFieldTitles(['Property','Value']);
 15         
 16     },
 17     
 18 
 19     // **************** PUBLIC METHODS **************** //
 20 
 21     
 22     /**
 23      * This function updates the table with data coming from
 24      * the currently selected element
 25      * @param element
 26      *          Something implementing TableModel interface
 27      *              * String getTitle()
 28      *              * Array getInfo()
 29      *              * Array getTriggerMapping(): Optional method
 30      */
 31     fillTable: function (/** Object */ element) {
 32         this._clearElement();
 33 
 34         var title = element.getTitle();
 35        
 36         this._propertiesTable.insertDataValues(element.getInfo());
 37         this._propertiesTable.setTitle((title ? 'Properties of '  + title : 'Properties'));
 38     },
 39 
 40     /**
 41      * This function add a new section to the table, without removing the actual
 42      * data. A section is a title row, and values
 43      */
 44     addSection: function (/** Array */ title, /** Hash */ values) {
 45         this._propertiesTable.insertFieldTitles(title);
 46         this._propertiesTable.insertDataValues(values);
 47     },
 48 
 49     // **************** PRIVATE METHODS **************** //
 50     
 51     /**
 52      * This function empties the table
 53      * @private
 54      */
 55     _clearElement: function (){
 56         this._propertiesTable.emptyTable();
 57         this._propertiesTable.setTitle('Properties');
 58         this._propertiesTable.insertFieldTitles(['Property','Value']);
 59     }
 60 
 61     
 62 });
