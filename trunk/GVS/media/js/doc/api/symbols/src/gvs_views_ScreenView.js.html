108     /**
109      * Colorize the component depending on the reachability
110      * @public @override
111      */
112     setReachability: function( /** Hash */ reachabilityData) {
113         
114         var satisfeable = reachabilityData.reachability;
115         
116         // screen
117         Utils.setSatisfeabilityClass(this.getNode(), satisfeable);
118        
119         // posts
120         this._postIcons.each(function(postIcon) {
121             Utils.setSatisfeabilityClass(postIcon.getNode(), satisfeable);  
122         });
123         
124         // pres
125         var preconditionList = reachabilityData.preconditions;  
126         if (preconditionList.length > 1) {
127             //More than one set of preconditions is NOT YET SUPPORTED
128             console.log("OR precondition support not implemented yet");
129             return;
130         } else {
131             var preconditionData = preconditionList[0];
132         }
133         
134         //Setting precondition reachability
135         var factFactory = FactFactorySingleton.getInstance();
136         var preReachability = new Hash();
137         $A(preconditionData).each(function(precondition) {
138             preReachability.set(factFactory.getFactUri(precondition), precondition.satisfied);    
139         });
140       
141         this._preIcons.each(function(preIcon) {
142             var factUri = preIcon.getFact().getUri();
143             Utils.setSatisfeabilityClass(preIcon.getNode(), preReachability.get(factUri));
144         });
145     },
