104     /**
105      * This function returns an array of lines representing the
106      * key information of the building block, in order to be shown in
107      * a table
108      * Implementing TableModel interface
109      * @type Hash
110      */
111     getInfo: function() {
112         var info = new Hash();
113 
114         info.set('Title', this.getTitle());
115         info.set('Description', this._buildingBlockDescription.description['en-gb']);
116         info.set('Tags', this._buildingBlockDescription.tags.collect(function(tag) {
117                 return tag.label['en-gb'];
118             }).join(", "));
119 
120         var params = document.createElement('div');
121         var paramsText = this.getParams();
122         var parameterized = paramsText.strip().length > 0;
123         if (parameterized) {
124             try {
125                 var json = json = cjson_parse(paramsText);
126 
127                 parameterized = false;
128                 for (var i in json) {
129                     params.addClassName('json-icon-bg');
130                     parameterized = true;
131                     break;
132                 }
133             } catch (e) {
134                 params.addClassName('text-icon-bg');
135             }
136         }
137 
138         if (!parameterized) {
139             var span = document.createElement('span');
140             span.addClassName('triggerInfo');
141             span.appendChild(document.createTextNode('No parameterized'));
142             params.appendChild(span);
143         }
144 
145         var paramsDialog = new ParamsDialog(this.getTitle(),
146                                this.getParams(),
147                                this._buildingBlockDescription.parameterTemplate,
148                                this.setParams.bind(this));
149         params.appendChild(paramsDialog.getButtonNode());
150 
151         info.set('Parameters', params);
152         return info;
153     },

377             this._view.addEventListener (function(event){

481     /**
482      * Gets a line of the list
483      * @private
484      * @type String
485      */
486     _getConditionItem: function(/** Object */ condition, /** Object */ reachability) {
487         var uri = FactFactory.getFactUri(condition);
488 
489         var fact = FactFactory.getFactIcon(condition, "embedded").getNode();
490         if (reachability.constructor == Hash) {
491             Utils.setSatisfeabilityClass(fact, reachability.get(uri));
492         } else {
493             Utils.setSatisfeabilityClass(fact, reachability);
494         }
495 
496         if (condition.positive === false) {
497             fact.addClassName('negative');
498         }
499 
500         var description = condition.label['en-gb'];
501 
502         return [fact, description, uri];
503     },

505     /*
506      * @private
507      * @type String
508      */
509     _buildDeleteUri: function() {
510         if (this.getId()) {
511             var originalParts = this.getId().split("/");
512             var parts = [
513                 originalParts[originalParts.length - 3],
514                 originalParts[originalParts.length - 2],
515                 originalParts[originalParts.length - 1]
516             ];
517             var uri = parts.join("/");
518             return URIs.catalogueDeleteCopy.replace("<id>", uri);
519         } else {
520             return "";
521         }
522     }
