 10          * List of open documents

 43 
 44     setEnabled: function(/** Boolean */ enabled) {
 45         if (enabled) {
 46             document.observe('keypress', this._onKeyPressed.bind(this));
 47         } else {
 48             document.stopObserving('keypress', this._onKeyPressed.bind(this));
 49         }        
 50     },

 53      * Creates a new screenflow document

 58     },
 59

 61      * Creates a new deployment document

 78     /**

 80      */
 81     addDocument: function(document){
 82         this._documents.set(document.getTabId(), document);
 83         dijit.byId("documentContainer").addChild(document.getTab());
 84         dijit.byId("documentContainer").selectChild(document.getTabId());
 85         $$(".tabLabel").each(function(canvas) {

 87                 var element = Event.element(e);
 88                 element.blur();
 89             })

147     /**
148      * Select Document event handler.
149      * @private
150      */
151     _selectDocument: function( /** {String|DOMNode} */ tab) {
152         var id;
153 
154         if (tab.id) {  // it is tab widget
155             id = tab.id;
156         } else { //it is a string id
157             id = tab;
158         }
159         this._currentDocument = this._documents.get(id);
160         
161         this._toolbar.setModel(1, this._currentDocument);
162     },
