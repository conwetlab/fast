1 var PlanSet = Class.create(BuildingBlockSet, /** @lends PlanSet.prototype */ {
  2     /**
  3      * This class handles a set of building blocks for a given context.
  4      * This list will be updated.
  5      * @constructs
  6      * @extends BuildingBlockSet
  7      */ 
  8     initialize: function($super) {
  9         $super([]);
 10         
 11         /** 
 12          * List of Plans
 13          * @type Array
 14          * @private @member
 15          */
 16         this._plans = new Array();
 17                
 18         
 19         this._factory = Catalogue.
 20             getBuildingBlockFactory(Constants.BuildingBlock.SCREEN);
 21     },
 22     
 23 
 24     // **************** PUBLIC METHODS **************** //
 25 
 26     
 27     /**
 28      * Returns all the building blocks ordered by plan from the set
 29      * @type Array
 30      * @override
 31      */
 32     getBuildingBlocks: function () {
 33         var result = new Array();
 34         this._plans.each(function(plan) {
 35             result.push(this._factory.getBuildingBlocks(plan));    
 36         }.bind(this));
 37         return result;
 38     },
 39     
 40     /**
 41      * Add new building blocks to the set by uri
 42      */
 43     setPlans: function (/** Array */ plans) {
 44         this._plans = plans.splice(0,10); //Just in case
 45         var uris = this._plans.flatten().uniq();
 46         this._factory.cacheBuildingBlocks(uris, this._cachedElements.bind(this));
 47     },
 48 
 49     // **************** PRIVATE METHODS **************** //
 50 
 51     /**
 52      * This is the callback called when returning from the 
 53      * building block factory
 54      */
 55     _cachedElements: function () {
 56         this._listener.setChanged();
 57     }
 58     
 59 });
