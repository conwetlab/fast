124         var onSuccess = function (/** XmlHttpRequest */ response){

126

129             this._lastName = remoteUser.user.last_name;

132             
133             if (this._ezWebURL[this._ezWebURL.length-1] != '/') {

135             }
136             
137             URIs.ezweb = this._ezWebURL;
138         }

143         var onError = function (/** XmlHttpRequest */ response, /** Exception */ e){         
144             Logger.serverError (response, e);
145         }
146               
147         PersistenceEngine.sendGet(URIs.userPreferences, this, onSuccess, onError);
148     },

160         }

162         /**

165          */
166         var onError = function (/** XmlHttpRequest */ response){

173             'email': this._email,
