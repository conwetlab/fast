 41      */

 43         if (!this._buildingBlockDescription.title) {
 44             this._updateTitle();

 49     /**
 50      * Creates the title of the palette component
 51      * @type String
 52      * @private
 53      */
 54     _updateTitle: function() {
 55         
 56         if (this._buildingBlockDescription['http://www.w3.org/2000/01/rdf-schema#label']) {
 57             this._buildingBlockDescription.title = 
 58                         this._buildingBlockDescription['http://www.w3.org/2000/01/rdf-schema#label'].
 59                         replace("@en","");      
 60 
 61         } else { //Extract the title from the uri
 62             var uri;
 63             if (this._buildingBlockDescription['uri']) {
 64                 uri = this._buildingBlockDescription['uri'];
 65             } else {
 66                 uri = Utils.extractURIfromPattern(this._buildingBlockDescription['pattern']);
 67             }
 68             this._buildingBlockDescription.title = this._createTitle(uri);
 69         }
 70     },
 71 
 72     
 73     /**
 74      * This function creates a title from the uri identifier
 75      * @private
 76      * @type String
 77      */
 78     _createTitle: function(/** String */ uri) {
 79         if (uri) {
 80             var pieces = uri.split("#");
 81             var identifier = "";
 82             if (pieces.length > 1){
 83                 identifier = pieces[1];               
 84             }
 85             else { //The uri has not identifier, try the last part of the url
 86                 pieces = uri.split("/");
 87                 identifier = pieces[pieces.length - 1];
 88             }
 89             return this._sanitizeTitle(identifier);
 90         } else {
 91             return "Unknown Domain Concept";
 92         }
 93     },
 94     
 95     /**
 96      * This function returns a human-readable title from an
 97      * identifier
 98      * @private
 99      * @type String
100      */
101     _sanitizeTitle: function (/** String */ title) {
102         //FIXME: I18NString
103         var result = title.replace("@en","");
104         //separate identifier from its forming words
105         var words = result.match(/[A-Z][a-z0-9]+|\s+[a-z][a-z0-9]*/g);
106         if (words) {
107             words.map(function(e) {
108                 return e.strip();
109             });
110             result = words.join(" ");
111         }
112         return result;       
113     }
