1 var Palette = Class.create(SetListener, /** @lends Palette.prototype */ {
  2 
  3     /**
  4      * Represents a palette of droppable components of a given type.
  5      *
  6      * @constructs
  7      * @extends SetListener
  8      */
  9     initialize: function(/** BuildingBlockSet */ set, /** DropZone */ dropZone,
 10             /** InferenceEngine */ inferenceEngine) {
 11         /**
 12          * @private @member
 13          * @type InferenceEngine
 14          */
 15         this._inferenceEngine = inferenceEngine;
 16 
 17         /**
 18          * Building block set
 19          * @type BuildingBlockSet
 20          * @private @member
 21          */
 22         this._set = set;
 23         
 24         /**
 25          * Zone to drop components
 26          * @type DropZone
 27          * @private @member
 28          */
 29         this._dropZone = dropZone;
 30         
 31         /**
 32          * Collection of components the palette offers.
 33          * @type Hash   Hash of URI to PaletteComponent
 34          * @private @member
 35          */
 36         this._components = new Hash();
 37 
 38         /**
 39          * Accordion pane node.
 40          * @type DOMNode
 41          * @private @member
 42          */
 43         this._node = null;
 44         
 45         /**
 46          * Palette content
 47          * @type DOMNode
 48          * @private @member
 49          */
 50         this._contentNode = null;
 51         
 52         this._renderUI(); 
 53         this._set.setListener(this);
 54     },
 55 
 56     // **************** PUBLIC METHODS **************** //
 57     
 58     /**
 59      * Starts data retrieval
 60      * @public
 61      */
 62     startRetrievingData: function() {
 63         this._set.startRetrievingData();
 64     },
 65 
 66     /**
 67      * Gets the node of the accordion pane
 68      * @type DOMNode
 69      * @public
 70      */
 71     getNode: function() {
 72         return this._node;
 73     },
 74     
 75     /**
 76      * This function will be called whenever
 77      * the set of building blocks changes
 78      * @overrides
 79      */
 80     setChanged: function () {
 81         this._updateComponents();
 82     },
 83 
 84     getBuildingBlockSet: function() {
 85         return this._set;
 86     },
 87     
 88     /**
 89      * All uris of all the components
 90      */
 91     getComponentUris: function() {
 92         var uris = [];
 93         this._set.getBuildingBlocks().each(function(buildingBlock) {
 94             uris.push({
 95                 uri: buildingBlock.uri
 96             });
 97         });
 98         return uris;
 99     },
100     
101     // **************** PRIVATE METHODS **************** //
102 
103     /**
104      * Creates the GUI stuff that shows the content: components and separators.
105      * @type DOMNode
106      * @private
107      */
108     _renderUI: function() {
109        
110         this._node = new dijit.layout.AccordionPane({
111             'title':this._set.getBuildingBlockName(),
112             'class':'paletteElement'
113         });
114         
115         this._contentNode = new Element('div', {
116             'class': 'paletteContent'
117         });
118         
119         this._node.setContent(this._contentNode);
120     },
121 
122     /**
123      * Updates the palette components from building blocks by querying its building block factory.
124      * @private
125      */
126     _updateComponents: function() {
127         var descs = this._set.getBuildingBlocks();
128   
129         $A(descs).each(
130             function(desc) {
131                 if (!this._components.get(desc.uri)) {
132                     this._addComponentFor(desc);
133                 }
134             }.bind(this)
135         );
136     },
137 
138     
139     /**
140      * Adds a new component to the palette
141      */
142     _addComponentFor: function(/** BuildingBlockDescription */ desc) {
143         var component = null;
144         
145         switch(this._set.getBuildingBlockType()) {
146             case Constants.BuildingBlock.SCREEN:
147                 component = new ScreenComponent(desc, this._dropZone, this._inferenceEngine);
148                 break;
149                 
150             case Constants.BuildingBlock.DOMAIN_CONCEPT:
151                 component = new DomainConceptComponent(desc, this._dropZone, this._inferenceEngine);
152                 break;
153                 
154             default:
155                 throw "Unsupported building block type";
156         }
157         this._components.set(desc.uri, component);
158 
159         var separator = new Element("div", {"class": "paletteSeparator"});   
160         this._contentNode.appendChild(component.getNode());
161         this._contentNode.appendChild(separator);
162     }
163 });
