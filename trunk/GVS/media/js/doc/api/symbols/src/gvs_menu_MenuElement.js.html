 26 var MenuElement = Class.create( /** @lends MenuElement.prototype */ {
 27     /**
 28      * Element for a menu
 29      * @constructs
 30      * @abstract
 31      */
 32     initialize: function(/** Number */ weight) {
 33         /**
 34          * The button Widget
 35          * @type dijit.*
 36          * @private
 37          */
 38         this._widget = null;
 39 
 40         /**
 41          * Ordering weight, the lower, the sooner the menu element
 42          * appears within a menu group
 43          * @type Number
 44          * @private
 45          */
 46         this._weight = weight;
 47         if (!this._weight) {
 48             this._weight = MenuElement.MAXIMUM_WEIGHT;
 49         }
 50     },
 51 
 52 
 53     // **************** PUBLIC METHODS **************** //
 54 
 55 
 56     /**
 57      * Returns the  widget
 58      * @type dijit.*
 59      */
 60     getWidget: function () {
 61         return this._widget;
 62     },
 63 
 64     /**
 65      * Returns the weight
 66      * @type Number
 67      */
 68     getWeight: function() {
 69         return this._weight;
 70     },
 71 
 72     /**
 73      * Register key handlers
 74      * @abstract
 75      */
 76     register: function(/** KeyPressRegistry */ registry) {
 77         throw "Abstract method invocation. MenuElement::register";
 78     },
 79 
 80     /**
 81      * Unregister key handlers and destroy the
 82      * widgets when necessary
 83      * @abstract
 84      */
 85     unregister: function(/** KeyPressRegistry */ registry) {
 86         throw "Abstract method invocation. MenuElement::unregister";
 87     }
 88 });
 89 // *************** CONSTANTS **************//
 90 MenuElement.MAXIMUM_WEIGHT = 10000;
