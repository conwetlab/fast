1 var KeyPressRegistry = Class.create( /** @lends KeyPressRegistry.prototype */ {
  2     /**
  3      * This class handles the different
  4      * onKeyPressed listeners, and the enabling and disabling of these events
  5      * @constructs
  6      */ 
  7     initialize: function() {
  8         /**
  9          * Hash of registered handlers
 10          * @type Hash
 11          * @private
 12          */
 13         this._handlers = new Hash();
 14         
 15         /**
 16          * The Keypress events are enabled
 17          */
 18         this._enabled = true;
 19         
 20         this.setEnabled(this._enabled);
 21     },
 22     
 23 
 24     // **************** PUBLIC METHODS **************** //
 25 
 26     
 27     /**
 28      * This function adds a listener to a specific key 
 29      * or key combination.
 30      */
 31     addHandler: function (/** String  */ accelKey, /** Function */ handler) {
 32         var key = accelKey.toLowerCase();
 33         if (this._handlers.get(key)) {
 34             shortcut.remove(key);
 35         }
 36         this._handlers.set(key, handler);
 37         shortcut.add(key, this._executeHandler.bind(this), {'disable_in_input': true});
 38     },
 39     
 40     /**
 41      * Asks if a key stroke is being used
 42      * @type Boolean
 43      */
 44     isRegistered: function(/** String */ key) {
 45         return (this._handler.get(key) ? true : false);    
 46     },
 47     
 48     /**
 49      * This function removes the handler for a key combination
 50      */
 51     removeHandler: function(/** String */ accelKey) {
 52         var key = accelKey.toLowerCase();
 53         this._handlers.unset(key);
 54         //shortcut.remove(key);
 55     },
 56     
 57     /**
 58      * This function enables or disables the onkeypress events
 59      */
 60     setEnabled: function(/** Boolean */ enabled) {
 61         this._enabled = enabled;       
 62     },
 63     
 64     // ********************** PRIVATE METHODS ********************* //
 65     
 66     /**
 67      * This function actually receives the keypress events and calls 
 68      * the handlers when necessary
 69      * @private
 70      */
 71     _executeHandler: function(/** Event */ e, /** String */ key) {
 72         if (this._enabled && this._handlers.get(key)) {
 73             this._handlers.get(key)();
 74         } 
 75     }
 76 });
