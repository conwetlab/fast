 18      /**
 19      * Creates a body to be sent in an AJAX call to the 
 20      * catalogue
 21      * @private
 22      * @overrides
 23      * @type String
 24      */
 25     _constructBody: function(/**Array*/ canvas, /** Object */ elements,
 26                     /** Array */ tags,
 27                     /** String*/ criteria, /** String(Optional) */ _method) {
 28         var method = Utils.variableOrDefault(_method, "");
 29         /*var domainContext = {
 30             'tags': tags,
 31             'user': GVS.getUser().getUserName()
 32         };*/
 33 
 34         var domainContext = {
 35             'tags': tags.collect(function(tag){ return tag.label['en-gb']}),
 36             'user': GVS.getUser().getUserName()
 37         }
 38         var body = {
 39             'canvas': canvas,
 40             'domainContext': domainContext,
 41             'criterion': criteria
 42         };
 43         if (method == "check") {
 44             body.search = false;
 45         }
 46         body = Object.extend(body, elements);
 47         return Object.toJSON(body);
 48     },
 49     
 50     /**
 51      * Gets the uri for a given operation
 52      * @private
 53      * @overrides
 54      */ 
 55     _getUri:function (/** String */ operation) {
 56         return URIs.catalogueScreenFindCheck;
 57     },
 58     
 59     /** 
 60      * onSuccess callback
 61      * @private
 62      * @overrides
 63      */
 64     _findCheckOnSuccess: function(/** XMLHttpRequest */ transport){
 65         var result = JSON.parse(transport.responseText);
 66         
 67         if (result.canvas && result.canvas.length > 0) {
 68             // There is some reachability information
 69               this.mine._updateReachability(result.canvas);
 70         }
 71         
 72         this.callback(result);        
 73     },
 74     
 75     /**
 76      * onSuccess callback
 77      * @private
 78      * @overrides
 79      */
 80     _checkOnSuccess: function(transport){
 81         try {
 82              var result = JSON.parse(transport.responseText);
 83              this.mine._updateReachability(result.canvas);
 84              this.callback(result);
 85         }
 86         catch (e) {
 87 
 88         }
 89     }
