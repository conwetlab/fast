1 var MenuAction = Class.create(MenuElement, /** @lends MenuAction.prototype */ {
  2     /**
  3      * Element for a menu
  4      * @constructs
  5      * @extends MenuElement
  6      * @param Hash extraParams (optional)
  7      *           Hash containing some of these extra params
  8      *           * (String)iconName Name of the icon if present
  9      *           * (Boolean)enabled state of the element
 10      *           * (String)accelKey key shortcut to access to the element
 11      */
 12     initialize: function($super, /** Object */ data, /** Boolean */ mainMenu) {
 13         $super(data.weight);
 14         if (mainMenu) {
 15             this._widget = new dijit.MenuBarItem ({
 16                 'label': data.label,
 17                 'onClick': data.handler,
 18                 'showLabel': true
 19             });
 20         } else {
 21             this._widget = new dijit.MenuItem ({
 22                 'label': data.label,
 23                 'onClick': data.handler,
 24                 'showLabel': true
 25             });
 26         }
 27 
 28         /**
 29          * Handler function
 30          * @type Function
 31          * @private
 32          */
 33         this._handler = data.handler;
 34 
 35         /**
 36          * Accel key shortcut to use the menu item,
 37          * if any
 38          * @type String
 39          * @private
 40          */
 41         this._shortcut = (data.shortcut ? data.shortcut : null);
 42 
 43         if (this._shortcut) {
 44             this._widget.attr('accelKey', this._shortcut);
 45         }
 46 
 47         /**
 48          * Item state
 49          * @type Boolean
 50          * @private
 51          */
 52         this._enabled = null;
 53         this.setEnabled(data.enabled !== undefined ? data.enabled : true);
 54 
 55         if (data.iconName) {
 56             this._widget.attr('iconClass', 'dijitMenuItemIcon menuIcon ' + data.iconName + 'MenuIcon');
 57         }
 58     },
 59 
 60 
 61     // **************** PUBLIC METHODS **************** //
 62 
 63 
 64     /**
 65      * Sets the element enabled
 66      * It must be called to set it disabled
 67      * when the menuitem is going to be left
 68      * outside of the menu
 69      */
 70     setEnabled: function(/** Boolean */ enabled) {
 71         this._enabled = enabled;
 72         this._widget.attr('disabled',!enabled);
 73     },
 74 
 75     /**
 76      * Register key handlers
 77      * @override
 78      */
 79     register: function(/** KeyPressRegistry */ registry) {
 80         if (this._shortcut) {
 81             registry.addHandler(this._shortcut, this._keyPressHandler.bind(this));
 82         }
 83     },
 84 
 85     /**
 86      * Unregister key handlers
 87      * @override
 88      */
 89     unregister: function(/** KeyPressRegistry */ registry) {
 90         if (this._shortcut) {
 91             registry.removeHandler(this._shortcut);
 92         }
 93     },
 94 
 95     // ************** PRIVATE METHODS **************** //
 96 
 97     /**
 98      * function to capture key strokes and decide if
 99      * calling the handler depending on the status of the
100      * action (enabled or disabled)
101      */
102     _keyPressHandler: function() {
103         if (this._enabled) {
104            this._handler();
105         }
106     }
107 });
