1 var Toolbar = Class.create( /** @class @lends Toolbar.prototype */ {
  2     /**
  3      * The toolbar itself
  4      *
  5      * @constructs
  6      * @constructor
  7      */
  8     initialize: function() {
  9         /**
 10          * @type dijit.dialog.Toolbar
 11          * @private @member
 12          */
 13         this._toolbar = dijit.byId("toolbar");
 14 
 15         /**
 16          * List of toolbar models
 17          * @type Array
 18          * @private @member
 19          */
 20         this._models = new Array();
 21 
 22         /**
 23          * List of section elements.
 24          * Each section have another array with the Widget of the buttons plus a
 25          * separator on the first position (index 0).
 26          *
 27          * @type Array
 28          * @private @member
 29          */
 30         this._sections = new Array();
 31         this._sections[0] = new Array();
 32     },
 33 
 34     /**
 35      * Sets the model for the toolbar section on the given position.
 36      * ToolbarModel must provide objects implementing the following interface:
 37      *
 38      *   interface ToolbarElement
 39      *      Widget getWidget()  // dojo widget
 40      * @
 41      */
 42     setModel: function(/** Number */position, /** ToolbarModel */ model) {
 43         this._removeModel(position);
 44 
 45         if (model) {
 46             var toolbarElements = model.getToolbarElements();
 47 
 48             if (toolbarElements.size() > 0) {
 49                 this._models[position] = model;
 50 
 51                 this._initSection(position);
 52 
 53                 toolbarElements.each(function (element) {
 54                     this._sections[position].push(element.getWidget());
 55                 }.bind(this));
 56             }
 57         }
 58 
 59         this._refreshToolbar();
 60     },
 61 
 62     // ************************ PRIVATE METHODS ************************* //
 63 
 64     /**
 65      * Removes the model from the given position
 66      * @private
 67      */
 68     _removeModel: function(/** Number */ position) {
 69         if (this._models[position]) {
 70             this._models[position] = null;
 71             this._sections[position].clear();
 72         }
 73     },
 74 
 75     /**
 76      * Initializes a section
 77      * @private
 78      */
 79     _initSection: function(/** Number */ position) {
 80         if (!this._sections[position]) {
 81             this._sections[position] = new Array();
 82         }
 83         if (position != 0 && !this._sections[position][0]) {
 84             this._sections[position][0] = new dijit.ToolbarSeparator();
 85         }
 86     },
 87 
 88     /**
 89      * Refresh the toolbar interface
 90      * Called when something happens (a model has been added or removed)
 91      * @private
 92      */
 93     _refreshToolbar: function() {
 94         this._toolbar.getDescendants().each(function(descendant) {
 95             this._toolbar.removeChild(descendant);
 96         }.bind(this));
 97 
 98         this._sections.each(function (section) {
 99             section.each(function (element) {
100                 this._toolbar.addChild(element);
101             }.bind(this));
102         }.bind(this));
103     }
104 
105 });
