 26 var FormDialog = Class.create( /** @lends FormDialog.prototype */ {
 27 
 28     /**
 29      * Dialog class
 30      * This creates a modal dialog with three zones:
 31      *     * headerNode: containing the title
 32      *     * contentNode: containing all the fields of the form
 33      *     * buttonsNode: containing the different buttons: handled by this class
 34      * @constructs
 35      * @param properties Hash
 36      * @param _options Hash(Optional)
 37      *         * (String) buttonPosition: Sets the layout of the dialog,
 38      *         stablishing the position of the button zone (top, bottom, left,
 39      *         right). Default: FormDialog.POSITION_BOTTOM
 40      *         * (Boolean) createMessageZone: Adds a new zone "messageNode" that
 41      *         can be used to show dinamic messages. Default: false
 42      *         * (Numeric) minMessageLines: This option allows you to define a
 43      *         minimum number of lines to reserve for the messageNode zone. This
 44      *         option is ignored if createMessageZone == false. Default: 1
 45      * @abstract
 46      */
 47     initialize: function(properties, _options) {
 48         _options = Utils.variableOrDefault(_options, {});
 49         this._options = Object.extend ({
 50             'buttonPosition': FormDialog.POSITION_BOTTOM,
 51             'createMessageZone': false,
 52             'minMessageLines': 1,
 53             'closable': true
 54         }, _options);
 55 
 56 
 57         var position = this._options.buttonPosition;
 58         this._dialog = new dijit.Dialog(properties);
 59 
 60         if (!this._options.closable) {
 61             this._dialog.closeButtonNode.style.display = 'none';
 62         }
 63 
 64         this._headerNode = new Element ('div',{
 65             'class': 'dialogHeader'
 66         });
 67 
 68         this._contentNode = new Element ('div',{
 69             'class': 'dialogContent'
 70         });
 71         this._messageNode = new Element ('div',{
 72             'class': 'dialogMessageZone hidden'
 73         });
 74         var messageWrapper = new Element ('div',{
 75             'class': 'dialogMessageWrapper'
 76         });
 77         messageWrapper.appendChild(this._messageNode);
 78         this._buttonNode = new Element ('div',{
 79             'class': 'dialogButtonZone'
 80         });
 81         this._formWidget = null;
 82 
 83         var containerDiv = new Element ('div', {
 84             'class': position
 85         });
 86 
 87         containerDiv.appendChild (this._headerNode);
 88         switch (position) {
 89             case FormDialog.POSITION_TOP:
 90                 containerDiv.appendChild (this._buttonNode);
 91 
 92                 if (this._options.createMessageZone)
 93                     containerDiv.appendChild (messageWrapper);
 94 
 95                 containerDiv.appendChild (this._contentNode);
 96                 break;
 97             default:
 98                 containerDiv.appendChild (this._contentNode);
 99 
100                 if (this._options.createMessageZone)
101                     containerDiv.appendChild (messageWrapper);
102 
103                 containerDiv.appendChild (this._buttonNode);
104                 break;
105         }
106         messageWrapper.style.minHeight = ((this._options.minMessageLines * 18) + 2) + 'px';
107 
108         this._dialog.attr ('content', containerDiv);
109 
110         this._initialized = false;
111         dojo.connect(this._dialog,"hide", this._hide.bind(this));
112 
113     },
114 
115 
116     // **************** PUBLIC METHODS **************** //

118     /**
119      * Shows the dialog
120      * @public
121      */
122     show: function() {
123         if (!this._initialized) {
124             this._initDialogInterface();
125             this._initialized = true;
126         } else {
127             this._reset();
128         }
129         GVS.setEnabled(false);
130         this._dialog.show();
131     },
132 
133     /**
134      * Returns true if the dialog is visible
135      */
136     isVisible: function() {
137         return this._dialog.attr("open");
138     },
139

141     /**
142      * initDialogInterface
143      * This function creates the dom structure
144      * @private
145      * @abstract
146      */
147     _initDialogInterface: function(){
148         throw "Abstract method invocation FormDialog :: _initDialogInterface";
149     },
150 
151     _hide: function() {
152         GVS.setEnabled(true);
153 
154         // Dojo lets you move the dialog anywhere you like
155         // but does not restore the window scroll when the dialog is closed.
156         // Force window scroll restore.
157         document.documentElement.scrollTop = 0;
158     },

160     /**
161      * Gets the form node.
162      * @type DOMNode
163      * @private
164      */
165     _getForm: function() {
166         return this._formWidget.domNode;
167     },

169     /**
170      * Gets the form Widget
171      * @type dijit.form.Form
172      * @private
173      */
174     _getFormWidget: function() {
175         return this._formWidget;
176     },
177 
178     /**
179      * This function adds a button with an onclick handler
180      * @type dijit.form.Button
181      * @private
182      */
183     _addButton: function (/** String */ label, /** Function */ handler) {
184         var button = new dijit.form.Button({
185             'label': label,
186             onClick: handler
187         });
188 
189 
190         this._buttonNode.appendChild(button.domNode);
191         return button;
192     },
193 
194     /**
195      * Remove all buttons
196      * @private
197      */
198     _removeButtons: function() {
199         this._buttonNode.update("");
200     },
201 
202     /**
203      * This function sets the header and a subtitle if passed
204      * @private
205      */
206     _setHeader: function (/** String */ title, /** String */ subtitle){
207 
208         this._headerNode.update("");
209         var titleNode = new Element("h2").update(title);
210         this._headerNode.appendChild(titleNode);
211 
212         if (subtitle && subtitle != ""){
213             var subtitleNode = new Element("div", {
214                 "class": "line"
215             }).update(subtitle);
216             this._headerNode.appendChild(subtitleNode);
217         }
218     },
219 
220     /**
221      * This function sets the dialog title
222      * @private
223      */
224     _setTitle: function (/** String */ title){
225         this._dialog.setAttribute('title', title);
226     },
227 
228     /**
229      * This function set the form content based on a array-like
230      * structure, containing the different elements of the form,
231      * and, optionally, form parameters
232      * @private
233      */
234     _setContent: function (/** Array | DOMNode */ data, /** Hash */ formParams){
235         if (data instanceof Array) {
236             // Form
237             if (formParams){
238                 this._formWidget = new dijit.form.Form(formParams);
239             } else {
240                 this._formWidget = new dijit.form.Form ({
241                     'method': 'post'
242                 });
243             }
244 
245             var outputNode = this._formWidget.domNode;
246 
247             // Instantiate form elements
248             $A(data).each (function(line){
249                 var lineNode;
250                 var inputNode;
251 
252                 switch (line.type) {
253                     case 'title':
254                         lineNode = new Element ('h3').update(line.value);
255                         break;
256 
257                     case 'input':
258                         if (line.regExp || line.required) {
259                             var input = new dijit.form.ValidationTextBox({
260                                             'name' : line.name,
261                                             'value': line.value,
262                                             'regExp': (line.regExp) ? line.regExp : '.*',
263                                             'required': (line.required) ? line.required : false,
264                                             'invalidMessage': (line.message) ? line.message : 'This field cannot be blank'
265                                         });
266                         } else {
267                             var input = new dijit.form.TextBox({
268                                             'name' : line.name,
269                                             'value': line.value
270                                         });
271                         }
272                         if (line.disabled) {
273                             input.attr('disabled', line.disabled);
274                         }
275                         inputNode = input.textbox;
276                         lineNode = this._createLine(line.label, input.domNode);
277                         break;
278 
279                     case 'checkbox':
280 
281                         var checkbox = new dijit.form.CheckBox({
282                                     'name' : line.name,
283                                     'value': line.value
284                                 });
285                         if (line.disabled) {
286                             checkbox.attr('disabled', line.disabled);
287                         }
288 
289                         if (line.checked) {
290                             checkbox.attr('checked', line.checked);
291                         }
292 
293                         inputNode = checkbox.domNode;
294                         lineNode = this._createLine(line.label, checkbox.domNode);
295                         break;
296 
297                     case 'label':
298                         lineNode = new Element('div', {
299                                         'class': 'line',
300                                         'style': line.style
301                                     }).update(line.value);
302                         break;
303 
304                     case 'pre':
305                         lineNode = new Element('pre', {
306                                         'class': 'line',
307                                         'style': line.style
308                                     }).update(line.value);
309                         break;
310 
311                     case 'hidden':
312                         lineNode = new Element('input',{
313                                         'type': 'hidden',
314                                         'name': line.name,
315                                         'value': line.value
316                                     });
317                         break;
318 
319                     case 'comboBox':
320                         inputNode = new Element('select', {
321                             'name': line.name
322                         });
323 
324                         $A(line.options).each(function(option) {
325                             var optionNode = new Element('option', {
326                                  'value': option.value
327                             }).update(option.label);
328 
329                             if (option.value == line.value) {
330                                 optionNode.selected = "selected";
331                             }
332 
333                             inputNode.appendChild(optionNode);
334                         });
335 
336                         lineNode = this._createLine(line.label, inputNode);
337                         break;
338 
339                     case 'textarea':
340                         var textArea = new dijit.form.SimpleTextarea({
341                             'name': line.name
342                         });
343                         textArea.attr("value", line.value);
344 
345                         inputNode = textArea.domNode;
346                         lineNode = this._createLine(line.label, inputNode);
347                         break;
348 
349                     case 'advancedSeparator':
350                         if (outputNode == this._formWidget.domNode) {
351                             var wrapper = new Element("div", {
352                                 "style": "display:none;"
353                             });
354 
355                             var link = new Element("a", {
356                                 "href": "javascript:"
357                             }).update("Show advanced options...");
358                             link.observe("click", function(e) {
359                                 if (wrapper.style.display == "none") {
360                                     wrapper.setStyle({"display":"block"});
361                                     link.update("Hide advanced options...");
362                                 } else {
363                                     wrapper.setStyle({"display":"none"});
364                                     link.update("Show advanced options...");
365                                 }
366                             });
367                             var linkLine = this._createLine("", link);
368                             this._formWidget.domNode.appendChild(linkLine);
369                             this._formWidget.domNode.appendChild(wrapper);
370                             outputNode = wrapper;
371                         } else {
372                             throw "It cannot be more than one advancedSeparator." +
373                             "FormDialog::_setContent";
374                         }
375                         break;
376 
377                     default:
378                         throw "Unimplemented form field type. " +
379                         "FormDialog::_setContent";
380                 }
381                 if (lineNode) {
382                     outputNode.appendChild(lineNode);
383                     lineNode = null;
384                 }
385                 if (inputNode) {
386                     this._armEvents(inputNode, line.events);
387                 }
388             }.bind(this));
389 
390             this._contentNode.update(this._formWidget.domNode);
391 
392         } else {
393             // Data is a DOMNode
394             this._contentNode.update(data);
395         }
396 
397         // Just in case
398         dojo.parser.parse(this._contentNode);
399     },
400 
401     /**
402      * Sets the message area content.
403      *
404      * @private
405      */
406     _setMessage: function(/** String */ message, type) {
407         if (arguments.length == 0) {
408             this._messageNode.className = 'dialogMessageZone hidden';
409             this._messageNode.update(' ');
410             return;
411         }
412 
413         this._messageNode.className = 'dialogMessageZone';
414         switch (type) {
415             case FormDialog.MESSAGE_WARNING:
416                 this._messageNode.addClassName('warning');
417                 break;
418             case FormDialog.MESSAGE_ERROR:
419                 this._messageNode.addClassName('error');
420                 break;
421             default:
422         }
423 
424         this._messageNode.innerHTML = message;
425     },
426 
427     /**
428      * Construct a form line provided the label text and the input node.
429      * @type DOMNode
430      * @private
431      */
432     _createLine: function(/** String */ label, /** DOMNode */ inputNode) {
433         var lineNode = new Element('div', {
434             'class' : 'line'
435         });
436         if (label) {
437             var labelNode = new Element ('label').update(label);
438             lineNode.appendChild(labelNode);
439         }
440         lineNode.appendChild(inputNode);
441         return lineNode;
442     },
443 
444 
445     /**
446      * Attach the event handlers to the input DOM node
447      * @private
448      */
449     _armEvents: function(/** DOMNode */ input, /** Hash */ events) {
450         $H(events).each(function(pair) {
451             Element.observe(input, pair.key, pair.value);
452         });
453     },
454 
455     /**
456      * This method is called to validate user input
457      * Overwrite when necessary.
458      * @private
459      */
460     _validate: function() {
461         return this._getFormWidget().validate();
462     },
463 
464     /**
465      * This method is called for reseting the dialog fields.
466      * Overload when necessary.
467      * @private
468      */
469     _reset: function() {
470         if (this._getFormWidget()) {
471             this._getFormWidget().validate();
472         }
473     }
474 
475 });
