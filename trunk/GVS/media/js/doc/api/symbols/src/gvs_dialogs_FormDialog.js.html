  3     /**
  4      * Dialog class
  5      * This creates a modal dialog with three zones:
  6      *     * headerNode: containing the title
  7      *     * contentNode: containing all the fields of the form
  8      *     * buttonsNode: containing the different buttons: handled by this class
  9      * @constructs
 10      * @param Hash params
 11      * @abstract
 12      */
 13     initialize: function(properties) {
 14         this._dialog = new dijit.Dialog(properties);
 15         
 16         this._headerNode = new Element ('div',{
 17             'class': 'dialogHeader'
 18         });
 19         
 20         this._contentNode = new Element ('div',{
 21             'class': 'dialogContent'
 22         });
 23         this._buttonNode = new Element ('div',{
 24             'class': 'dialogButtonZone' 
 25         });
 26         this._formWidget = null;
 27         
 28         
 29         var containerDiv = new Element ('div');
 30         
 31         
 32         containerDiv.insert (this._headerNode);
 33         containerDiv.insert (this._contentNode);
 34         containerDiv.insert (this._buttonNode);
 35         this._dialog.attr ('content', containerDiv);
 36         
 37         this._initialized = false;
 38     },

126     /**
127      * This function sets the header and a subtitle if passed
128      * @private
129      */
130     _setHeader: function (/** String */ title, /** String */ subtitle){
131         
132         var titleNode = new Element("h2").update(title);
133         this._headerNode.insert(titleNode);
134         
135         if (subtitle && subtitle != ""){
136             var subtitleNode = new Element("div", {
137                 "class": "line"
138             }).update(subtitle);
139             this._headerNode.insert(subtitleNode);
140         }
141     },
142 
143     /**
144      * This function set the form content based on a array-like
145      * structure, containing the different elements of the form,
146      * and, optionally, form parameters
147      * @private
148      */
149     _setContent: function (/** Array | DOMNode */ data, /** Hash */ formParams){
150         if (data instanceof Array) {
151             // Form
152             if (formParams){
153                 this._formWidget = new dijit.form.Form(formParams);
154             } else {
155                 this._formWidget = new dijit.form.Form ({
156                     'method': 'post'
157                 });
158             }
159             
160             // Instantiate form elements
161             $A(data).each (function(line){
162                 var lineNode;
163                 var inputNode;
164                 
165                 switch (line.type) {
166                     case 'title':
167                         lineNode = new Element ('h3').update(line.value);
168                         break;
169                         
170                     case 'input':
171                         if (line.regExp) {
172                             var input = new dijit.form.ValidationTextBox({
173                                             'name' : line.name,
174                                             'value': line.value,
175                                             'regExp': line.regExp,
176                                             'required': true,
177                                             'invalidMessage': line.message
178                                         });
179                         } else {
180                             var input = new dijit.form.TextBox({
181                                             'name' : line.name,
182                                             'value': line.value   
183                                         });
184                         }             
185                         inputNode = input.domNode;
186                         lineNode = this._createLine(line.label, inputNode);
187                         break;
188     
189                     case 'freeText': // FIXME: crappy name
190                         lineNode = new Element('div', {
191                                         'class': 'line'
192                                     }).update(line.value);
193                         break;
194                         
195                     case 'hidden':
196                         lineNode = new Element('input',{
197                                         'type': 'hidden',
198                                         'name': line.name,
199                                         'value': line.value
200                                     });
201                         break;                    
202                                      
203                     case 'comboBox':
204                         inputNode = new Element('select', {
205                             'name': line.name
206                         });
207                         
208                         $A(line.options).each(function(option) {
209                             var optionNode = new Element('option', {
210                                  'value': option.value
211                             }).update(option.label);
212                             
213                             if (option.value == line.value) {
214                                 optionNode.selected = "selected";
215                             }
216                             
217                             inputNode.appendChild(optionNode);
218                         });
219                         
220                         lineNode = this._createLine(line.label, inputNode);
221                         break;
222         
223                     //TODO: Implement more when necessary
224                         
225                     default:
226                         throw "Unimplemented form field type";
227                 }
228                       
229                 this._formWidget.domNode.appendChild(lineNode);   
230                 
231                 this._armEvents(inputNode, line.events);
232             }.bind(this));
233             
234             this._contentNode.update(this._formWidget.domNode);  
235                         
236         } else {
237             // Data is a DOMNode
238             this._contentNode.update(data);
239         }        
240 
241         //Just in case
242         dojo.parser.parse(this._contentNode);
243     },

246     /**
247      * Construct a form line provided the label text and the input node.
248      * @type DOMNode
249      * @private
250      */
251     _createLine: function(/** String */ label, /** DOMNode */ inputNode) {
252         var lineNode = new Element('div', {
253                         'class' : 'line'
254                     });
255         var labelNode = new Element ('label').update(label);
256         lineNode.appendChild(labelNode);
257         lineNode.appendChild(inputNode);
258         
259         return lineNode;
260     },

264      * Attach the event handlers to the input DOM node

267     _armEvents: function(/** DOMNode */ input, /** Hash */ events) {
268         $H(events).each(function(pair) {
269             Element.observe(input, pair.key, pair.value);
270         });
271     },

281         }
