1 /**
  2  * <p>This class implements the Singleton Design Pattern to make sure there is 
  3  * only one instance of the class GVS.
  4  *
  5  * <p> It should be accessed as follows.
  6  *
  7  * @constructor
  8  * @example
  9  * var gvs = GVSSingleton.getInstance();
 10  */ 
 11 var GVSSingleton = function() {
 12 
 13     /**
 14      * Singleton instance
 15      * @private @member
 16      */
 17     var _instance = null;
 18     
 19 
 20     var GVS = Class.create(ToolbarModel,
 21         /** @lends GVSSingleton-GVS.prototype */ {
 22 
 23         /** 
 24          * GVS is the system facade.
 25          * @constructs
 26          * @extends ToolbarModel
 27          */
 28         initialize: function($super) {
 29             $super();
 30             
 31             /** 
 32              * @type DocumentController
 33              * @private @member
 34              */
 35             this._documentController = null;
 36 
 37             /** 
 38              * Hash keeping the action implementations.
 39              * @type Hash
 40              * @private @member
 41              */
 42             this._actions = null;
 43             
 44             /** 
 45              * User data object
 46              * @type User
 47              * @private @member
 48              */
 49             this._user = new User();
 50             
 51             /** 
 52              * Hash containing the different dialogs used in the welcome document
 53              * @type Hash
 54              * @private @member
 55              */            
 56             this._dialogs = new Hash();
 57             
 58             this._dialogs.set ("addScreen", new AddScreenDialog());
 59             this._dialogs.set ("newScreenflow", new NewScreenflowDialog());
 60             this._dialogs.set ("preferences", new PreferencesDialog());      
 61             
 62         },
 63         
 64 
 65         // **************** PUBLIC METHODS **************** //
 66         
 67 
 68         /**
 69          * Creates all the other objects and installs the event 
 70          * handlers.
 71          * @public
 72          */
 73         init: function () {
 74             this._actions = {
 75                 openScreenflow: this._openScreenflow.bind(this),
 76                 newScreenflow: this._newScreenflow.bind(this),
 77                 addScreen: this._addScreen.bind(this)
 78             };
 79             this._documentController = new DocumentController();
 80             
 81             /**
 82              * Toolbar buttons
 83              */
 84             this._addToolbarElement('home', new ToolbarButton(
 85                 'Home',
 86                 'home',
 87                 this._documentController.showWelcomeDocument.bind(this._documentController)
 88             ));
 89             
 90             var preferencesDialog = this._dialogs.get("preferences");
 91             this._addToolbarElement('preferences', new ToolbarButton(
 92                 'User Preferences',
 93                 'preferences',
 94                 preferencesDialog.show.bind(preferencesDialog)
 95             ));
 96             
 97             this._documentController.getToolbar().setModel(0, this);
 98             
 99             this._documentController.getMenu().setModel('GVS', this);
100         },
101 
102         /**
103          * Ask the GVS application to perform a high-level action.
104          * @public
105          */
106         action: function (/** String */ actionName){
107             if (this._actions[actionName]){
108                 // Execute the action
109                 this._actions[actionName]();
110             } else{
111                 throw "Unexpected exception: GVS::action";
112             }
113         },
114 
115         /**
116          * Gets the document controller
117          * @type DocumentController
118          * @public
119          */
120         getDocumentController: function () {
121             return this._documentController;
122         },
123         
124         /**
125          * Gets the user account object
126          * @type User
127          * @public
128          */
129         getUser: function () {
130             return this._user;
131         },
132         
133         /**
134          * Sets the platform itself enabled or disabled
135          * (for modal dialogs)
136          */
137         setEnabled: function (/** Boolean */ enabled) {
138            this._documentController.getKeyPressRegistry().setEnabled(enabled);
139         },
140         
141         /**
142          * Implementing MenuModel interface
143          * @type Object
144          */
145         getMenuElements: function() {        
146             return {};
147         },
148         
149         // **************** PRIVATE METHODS **************** //
150 
151         /** 
152          * High-level action for creating a new screenflow.
153          *
154          * @private
155          */
156         _newScreenflow: function (){
157             this._dialogs.get("newScreenflow").show();
158         },
159         
160         /** 
161          * High-level action for adding a new screen to the catalogue.
162          *
163          * @private
164          */
165         _addScreen: function(){
166             this._dialogs.get("addScreen").show();
167         },
168         /**
169          * Open a screenflow
170          * @private
171          */
172         _openScreenflow: function() {
173             alert("open");
174         }
175     });
176     
177   
178     return new function(){
179         /**
180          * Returns the singleton instance
181          * @type GVS
182          */
183         this.getInstance = function(){
184             if (_instance == null) {
185                 _instance = new GVS();
186             }
187             return _instance;
188         }
189     }
190 }();
