1 /**
  2  * <p>This class implements the Singleton Design Pattern to make sure there is 
  3  * only one instance of the class GVS.
  4  *
  5  * <p> It should be accessed as follows.
  6  *
  7  * @constructor
  8  * @example
  9  * var gvs = GVSSingleton.getInstance();
 10  */ 
 11 var GVSSingleton = function() {
 12 
 13     /**
 14      * Singleton instance
 15      * @private @member
 16      */
 17     var _instance = null;
 18     
 19 
 20     var GVS = Class.create(
 21         /** @lends GVSSingleton-GVS.prototype */ {
 22 
 23         /** 
 24          * GVS is the system facade.
 25          * @constructs
 26          */
 27         initialize: function() {
 28             /** 
 29              * @type DocumentController
 30              * @private @member
 31              */
 32             this._documentController = null;
 33 
 34             /** 
 35              * Hash keeping the action implementations.
 36              * @type Hash
 37              * @private @member
 38              */
 39             this._actions = null;
 40         },
 41         
 42 
 43         // **************** PUBLIC METHODS **************** //
 44         
 45 
 46         /**
 47          * Creates all the other objects and installs the event 
 48          * handlers.
 49          * @public
 50          */
 51         init: function () {
 52             this._actions = {
 53                 openScreenflow: function (){
 54                     alert("open");
 55                 },
 56                 newScreenflow: this._newScreenflow.bind(this)
 57             };
 58             this._paletteController = new PaletteController();
 59             this._documentController = new DocumentController();
 60         },
 61 
 62         /**
 63          * Ask the GVS application to perform a high-level action.
 64          * @public
 65          */
 66         action: function (/** String */ actionName){
 67             if (this._actions[actionName]){
 68                 // Execute the action
 69                 this._actions[actionName]();
 70             } else{
 71                 // TODO: Add error handling
 72                 alert ("Invalid action");
 73             }
 74         },
 75         
 76         /**
 77          * Gets the palette controller
 78          * @type PaletteController
 79          * @public
 80          */
 81         getPaletteController: function () {
 82             return this._paletteController;
 83         },
 84         
 85         // **************** PRIVATE METHODS **************** //
 86 
 87         /** 
 88          * High-level action for creating a new screenflow.
 89          *
 90          * @private
 91          */
 92         _newScreenflow: function (){
 93             var title = prompt("Choose Screenflow name", "NewScreenflow");
 94             if (title && title != ""){
 95                 var screenflow = new ScreenflowDocument(title);
 96                 this._documentController.addDocument(screenflow);   
 97             }
 98         }
 99         
100     });
101     
102   
103     return new function(){
104         /**
105          * Returns the singleton instance
106          * @type GVS
107          */
108         this.getInstance = function(){
109             if (_instance == null) {
110                 _instance = new GVS();
111             }
112             return _instance;
113         }
114     }
115 }();
