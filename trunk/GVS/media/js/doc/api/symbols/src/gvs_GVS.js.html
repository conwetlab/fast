 77     /**
 78      * Creates all the other objects and installs the event
 79      * handlers.
 80      * @public
 81      */
 82     init: function(){
 83 
 84         this._actions = {
 85             newBuildingBlock: this._newBuildingBlock.bind(this),
 86             newScreenflow: this._newScreenflow.bind(this),
 87             addScreen: this._addScreen.bind(this),
 88             newScreen: this._newScreen.bind(this),
 89             browseScreenflows: this._browseScreenflows.bind(this),
 90             browseScreens: this._browseScreens.bind(this),
 91             browseBuildingBlocks: this._browseBuildingBlocks.bind(this),
 92             showAbout: this._showAboutDialog.bind(this),
 93             wrapperService: this._openWrapperService.bind(this),
 94             mediation: this._openMediationTool.bind(this),
 95             manageConcepts: this._openFactTool.bind(this)
 96         };
 97 
 98         this._dialogs.set("addScreen", new AddScreenDialog());
 99         this._dialogs.set("newScreenflow", new NewBuildingBlockDialog('Screenflow'));
100         this._dialogs.set("newScreen", new NewBuildingBlockDialog('Screen'));
101         this._dialogs.set("newForm", new NewBuildingBlockDialog('Form'));
102         this._dialogs.set("newOperator", new NewBuildingBlockDialog('Operator'));
103         this._dialogs.set("newResource", new NewBuildingBlockDialog('Resource'));
104         this._dialogs.set("browseScreenflows", new ManageScreenflowsDialog());
105         this._dialogs.set("browseScreens", new ManageScreensDialog());
106         this._dialogs.set("browseBuildingBlocks", new ManageBuildingBlocksDialog());
107         this._dialogs.set("preferences", new PreferencesDialog());
108         this._dialogs.set("newBuildingBlock", new NewBuildingBlockCodeDialog());
109 
110         this._documentController = new DocumentController();
111 
112         // Toolbar
113         this._addToolbarButtons();
114 
115         // Menu
116         this._setMenuItems();
117 
118         this._documentController.getToolbar().setModel(0, this);
119 
120         this._documentController.getMenu().setModel('GVS', this);
121     },
122 
123     /**
124      * Ask the GVS application to perform a high-level action.
125      * @public
126      */
127     action: function(/** String */actionName){
128         if (this._actions[actionName]) {
129             // Execute the action
130             this._actions[actionName]();
131         }
132         else {
133             alert("This function is being implemented. Stay tuned");
134         }
135     },
136 
137     /**
138      * Gets the document controller
139      * @type DocumentController
140      * @public
141      */
142     getDocumentController: function(){
143         return this._documentController;
144     },
145 
146     /**
147      * Gets the user account object
148      * @type User
149      * @public
150      */
151     getUser: function(){
152         return this._user;
153     },
154 
155     /**
156      * Sets the platform itself enabled or disabled
157      * (for modal dialogs)
158      */
159     setEnabled: function(/** Boolean */enabled){
160         this._documentController.getKeyPressRegistry().setEnabled(enabled);
161     },
162 
163     /**
164      * Implementing MenuModel interface
165      * @type Object
166      */
167     getMenuElements: function(){
168         return this._menuConfig;
169     },

261         this._addToolbarElement('preferences', new ToolbarButton('User Preferences', 'preferences', preferencesDialog.show.bind(preferencesDialog)));
262     },
263 
264     /**
265      * Init the loading of the about section
266      * @private
267      */
268     _showAboutDialog: function() {
269         PersistenceEngine.sendGet(URIs.about, this, this._onSuccessAbout, Utils.onAJAXError);
270     },
271     /**
272      * OnSuccess handler
273      */
274     _onSuccessAbout: function (/** XMLHttpRequest */ transport) {
275         var dialog = new ExternalContentDialog("About GVS");
276         dialog.show(transport.responseText);
277     },
