 26 var ManageBuildingBlocksDialog = Class.create(GalleryDialog /** @lends ManageScreensDialog.prototype */, {
 27     /**
 28      * This class handles the dialog
 29      * to open an existing screen
 30      * @constructs
 31      * @extends GalleryDialog
 32      */
 33     initialize: function($super) {
 34         $super("Building blocks browser", {
 35             'onDblClick': this._openBuildingBlock.bind(this),
 36             'disableIfNotValid': true
 37         });
 38 
 39         /**
 40          * List of building blocks
 41          * @type Array
 42          * @private
 43          */
 44         this._buildingBlocks = null;
 45 
 46         this._showForms = true;
 47         this._showOperators = true;
 48         this._showResources = true;
 49     },
 50 
 51 
 52     // **************** PUBLIC METHODS **************** //
 53 
 54     /**
 55      * @override
 56      */
 57     show: function() {
 58         this._loadBuildinBlocks(this._show.bind(this));
 59     },
 60 
 61 
 62     // **************** PRIVATE METHODS **************** //
 63 
 64 
 65     /**
 66      * Create the search bar
 67      * @private
 68      * @type HTMLNode
 69      */
 70     _createSearchBar: function($super) {
 71         var searchBar = $super();
 72 
 73         var content = document.createElement('div');
 74         content.style.textAlign = 'right';
 75         searchBar.appendChild(content);
 76 
 77         var checkBox = new dijit.form.CheckBox({
 78             name: 'Forms',
 79             checked: this._showForms,
 80             onChange: function(b) {
 81                 this._showForms = !(this._showForms && (this._showOperators || this._showResources));
 82                 this._reload();
 83             }.bind(this)
 84         });
 85         content.appendChild(checkBox.domNode);
 86 
 87         var label = document.createElement('span');
 88         label.innerHTML = 'Forms ';
 89         content.appendChild(label);
 90 
 91         var checkBox = new dijit.form.CheckBox({
 92             name: 'Operators',
 93             checked: this._showOperators,
 94             onChange: function(b) {
 95                 this._showOperators = !(this._showOperators && (this._showForms || this._showResources));
 96                 this._reload();
 97             }.bind(this)
 98         });
 99         content.appendChild(checkBox.domNode);
100 
101         var label = document.createElement('span');
102         label.innerHTML = 'Operators ';
103         content.appendChild(label);
104 
105         var checkBox = new dijit.form.CheckBox({
106             name: 'Resources',
107             checked: this._showResources,
108             onChange: function(b) {
109                 this._showResources = !(this._showResources && (this._showForms || this._showOperators));
110                 this._reload();
111             }.bind(this)
112         });
113         content.appendChild(checkBox.domNode);
114 
115         var label = document.createElement('span');
116         label.innerHTML = 'Resources';
117         content.appendChild(label);
118 
119         return searchBar;
120     },
121 
122     /**
123      * initDialogInterface
124      * This function creates the dom structure
125      * @private
126      * @override
127      */
128     _initDialogInterface: function (){
129 
130         this._setHeader("Browse your building blocks",
131                 "These are the building blocks you have created. Here you can " +
132                 "continue editing them and share your work with the community");
133 
134         this._setFields([{
135                 'title': 'Icon',
136                 'className': 'icon'
137             }, {
138                 'title': 'Screen Name',
139                 'className': 'name'
140             }, {
141                 'title': 'Screen Version',
142                 'className': 'version'
143             }, {
144                 'title': 'Tags',
145                 'className': 'tags'
146             }, {
147                 'title': 'Description',
148                 'className': 'description'
149             }, {
150                 'title': 'Sharing',
151                 'className': 'sharing'
152             }
153         ]);
154 
155         this._setButtons([{
156                 'value': 'Open building block',
157                 'handler': this._openBuildingBlock.bind(this),
158                 'disableIfNotValid': true
159             }, {
160                 'value': 'Share/Unshare building block',
161                 'handler': this._shareBuildingBlock.bind(this)
162             }, {
163                 'value': 'Delete building block',
164                 'handler': this._deleteBuildingBlock.bind(this)
165             }
166         ]);
167 
168         this._createBuildingBlockList();
169         this._render();
170     },
171 
172     /**
173      * Creates the building block list to be handled by its parent class
174      * @private
175      */
176     _createBuildingBlockList: function() {
177         this._emptyRows();
178         this._buildingBlocks.each(function(screen) {
179             var valid = ! screen.uri;
180             this._addRow({
181                             'key': screen.id,
182                             'values': [
183                                 new Element('img', {'src': screen.icon}),
184                                     screen.name,
185                                 '<span class="bold">Version: </span>' +
186                                      screen.version,
187                                 '<span class="bold">Tags: </span>' +
188                                     screen.tags.collect(function(tag) {
189                                         return tag.label['en-gb'];
190                                     }).join(", "),
191                                 '<span class="bold">Description </span><br />'+
192                                     screen.description['en-gb'],
193                                 '<span class=' + (screen.uri ? '"shared"': '"unshared"') +
194                                     '> </span>'
195                              ],
196                              'isValid': valid
197                         });
198         }.bind(this));
199     },
200 
201     /**
202      * Open a building block by its id
203      * @private
204      */
205     _openBuildingBlock: function(/** String */ id) {
206         var documentController = GVS.getDocumentController();
207         documentController.loadBuildingBlock(id);
208         this._dialog.hide();
209     },
210 
211     /**
212      * Share or unshare a screen depending of its status
213      * @private
214      */
215     _shareBuildingBlock: function(/** String */ id) {
216         var uri = URIs.share.replace("<id>", id);
217 
218         var screen = this._buildingBlocks.detect(function(element) {
219             return element.id == id;
220         });
221         if (screen.uri) {
222             // Unshare screen
223             PersistenceEngine.sendDelete(uri, this, this._reload, Utils.onAJAXError);
224         } else {
225             // Share screen
226             PersistenceEngine.sendPost(uri, null, null, this, this._reload,
227                                         Utils.onAJAXError);
228         }
229     },
230 
231     /**
232      * Starts the deletion of a building block
233      * @private
234      */
235     _deleteBuildingBlock: function(/** String */ id) {
236         confirm("Are you sure you want to delete the building block? " +
237             "This action cannot be undone",
238             this._confirmDelete.bind({'mine': this, 'id': id}));
239     },
240 
241     /**
242      * Execute the deletion of the building block
243      * @private
244      */
245     _confirmDelete: function(value) {
246         var uri = URIs.buildingblock + this.id;
247         if (value) {
248             PersistenceEngine.sendDelete(uri, this.mine, this.mine._reload, Utils.onAJAXError);
249         }
250     },
251 
252     /**
253      * Reloads the building block list
254      * @private
255      */
256     _reload: function() {
257         this._loadBuildinBlocks(function(){
258             this._createBuildingBlockList();
259             this._render(false);
260         }.bind(this));
261     },
262 
263     /**
264      * Load selected building blocks
265      * @private
266      */
267     _loadBuildinBlocks: function(onSucess) {
268         var types = [];
269         this._buildingBlocks = [];
270 
271         if (this._showForms) {
272             types.push('form');
273         }
274         if (this._showOperators) {
275             types.push('operator');
276         }
277         if (this._showResources) {
278             types.push('resource');
279         }
280         this._getBuildingBlocks(types, this, onSucess);
281     },
282 
283     /**
284      * Get selected building blocks
285      * @private
286      */
287     _getBuildingBlocks: function(/*Array<String>*/types, /*Object*/context, /*Function*/onSucess) {
288         var type = types.pop();
289         if (!type && onSucess instanceof Function) {
290             onSucess.call(context);
291         } else {
292             PersistenceEngine.sendGet(URIs[type], context, function(transport) {
293                 this._buildingBlocks = this._buildingBlocks.concat(JSON.parse(transport.responseText));
294                 this._getBuildingBlocks(types, context, onSucess);
295             }, Utils.onAJAXError);
296         }
297     }
298 
299 });
