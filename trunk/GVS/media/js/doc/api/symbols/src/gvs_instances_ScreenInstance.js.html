1 var ScreenInstance = Class.create(ComponentInstance,
  2     /** @lends ScreenInstance.prototype */ {
  3 
  4     /**
  5      * Screen instance.
  6      * @constructs
  7      * @extends ComponentInstance
  8      */
  9     initialize: function($super, /**BuildingBlockDescription*/ buildingBlockDescription,
 10              /** InferenceEngine */ inferenceEngine) {
 11         $super(buildingBlockDescription, inferenceEngine);
 12 
 13         /**
 14          * @type PreviewDialog
 15          * @private @member
 16          */
 17         this._dialog = null;
 18 
 19         this._menu.addOption('Preview', function(){
 20             this.showPreviewDialog();
 21         }.bind(this));
 22         if (this.getBuildingBlockDescription().definition != null) {
 23             this._menu.addOption('Clone', function(){
 24                 this.document.cloneElement(this);
 25             }.bind(this));
 26         }
 27         this._menu.addOption('Create a Plan', function(){
 28             this.document.getPlansElement(this);
 29         }.bind(this));
 30 
 31         /**
 32          * Screen caption (optional
 33          * @private
 34          * @type String
 35          */
 36         this._caption = null;
 37     },
 38 
 39     // **************** PUBLIC METHODS **************** //
 40 
 41     /**
 42      * @override
 43      */
 44     getInfo: function() {
 45         var info = new Hash();
 46         var titleDialog = new TitleDialog(this.getTitle(),
 47                                           this.setTitle.bind(this));
 48         var titleArea = new Element("div");
 49         var titleText = new Element("span").update(this.getTitle());
 50         titleArea.appendChild(titleText);
 51         titleArea.appendChild(titleDialog.getButtonNode());
 52         info.set('Title', titleArea);
 53 
 54         info.set('Description', this._buildingBlockDescription.description['en-gb']);
 55 
 56         var captionDialog = new CaptionDialog(this._caption,
 57                                           this.setCaption.bind(this));
 58         var captionArea = new Element("div");
 59         var captionText;
 60         if (this._caption != null) {
 61             captionText = new Element("span");
 62             var showedText = this._caption.stripTags().substring(0,10);
 63             if (showedText != this._caption.stripTags()) {
 64                 // The caption is longer than 10 chars
 65                 showedText += "...";
 66             }
 67             captionText.update(showedText);
 68         } else {
 69             captionText = new Element("span", {
 70                 "class":"triggerInfo"
 71             }).update("No caption set");
 72         }
 73         captionArea.appendChild(captionText);
 74         captionArea.appendChild(captionDialog.getButtonNode());
 75         info.set('Screen caption', captionArea);
 76 
 77         info.set('Tags', this._buildingBlockDescription.tags.collect(function(tag) {
 78                 return tag.label['en-gb'];
 79             }).join(", "));
 80         return info;
 81     },
 82 
 83     /**
 84      * Returns the caption of the screen if any
 85      */
 86     getCaption: function() {
 87         return this._caption;
 88     },
 89 
 90     /**
 91      * sets the caption
 92      */
 93     setCaption: function(/** String  */ caption) {
 94         this._caption = caption.stripScripts();
 95 
 96         if (this._listener && this._listener.modified) {
 97             this._listener.modified(this);
 98         }
 99     },
100 
101 
102     /**
103      * This function shows the dialog to change
104      * the instance properties
105      */
106     showPreviewDialog: function () {
107         if (!this._dialog) {
108             this._dialog = new PreviewDialog(this.getTitle(), this._buildingBlockDescription.getPreview());
109         }
110         this._dialog.show();
111     },
112 
113     // **************** PRIVATE METHODS **************** //
114     /**
115      * Creates a new View instance for the component
116      * @type BuildingBlockView
117      * @override
118      */
119     _createView: function () {
120         return new ScreenView(this._buildingBlockDescription);
121     },
122 
123     /**
124      * This function is called when the attached view is dbl-clicked
125      * @private
126      * @override
127      */
128     _onDoubleClick: function (/** Event */ event){
129         this.showPreviewDialog();
130     }
131 });
