 26 var DomainConceptFactory = Class.create(BuildingBlockFactory,
 27     /** @lends DomainConceptFactory.prototype */ {
 28 
 29     /**
 30      * Factory of domain concept building blocks.
 31      * @constructs
 32      * @extends BuildingBlockFactory
 33      */
 34     initialize: function($super) {
 35         $super();
 36 
 37     },
 38 
 39     /**
 40      * Gets the type of building block this factory mades.
 41      * @type String
 42      * @override
 43      */
 44     getBuildingBlockType: function (){
 45         return Constants.BuildingBlock.DOMAIN_CONCEPT;
 46     },
 47 
 48     // **************** PUBLIC METHODS **************** //
 49     /**
 50      * @override
 51      */
 52     getBuildingBlocks: function(/** Array */ tags, /** Function */ callback){
 53         var url = this._createUrl(tags);
 54         PersistenceEngine.sendGet(url,
 55                 {
 56                     'callback': callback
 57                 },
 58                 this._onSuccess, this._onError);
 59     },
 60 
 61     // **************** PUBLIC METHODS **************** //
 62 
 63     _createUrl: function(/** Array */ tags){
 64         if (tags.size() > 0) {
 65             var processedTags = tags.collect(function(tag) {
 66                 return tag.label['en-gb'];
 67             }).join("+");
 68             return URIs.catalogueTagConcepts.replace('<tags>', processedTags);
 69         } else {
 70             return URIs.catalogueAllConcepts;
 71         }
 72     },
 73 
 74     _onSuccess: function (/** XMLHttpRequest */ transport) {
 75         var metadata = transport.responseText.evalJSON();
 76         var result = new Array();
 77 
 78         $A(metadata).each(function(conceptProperties) {
 79             result.push(new PrePostDescription(conceptProperties));
 80         });
 81         this.callback(result);
 82     },
 83 
 84     _onError: function (/**XMLHttpRequest*/ transport, /** Exception */ e) {
 85         Logger.serverError(transport, e);
 86     }
 87 
 88 });
