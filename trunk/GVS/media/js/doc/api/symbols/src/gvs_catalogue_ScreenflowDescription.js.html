1 var ScreenflowDescription = Class.create(BuildingBlockDescription,
  2     /** @lends ScreenflowDescription.prototype */ {
  3 
  4     /**
  5      * Screenflow building block description.
  6      * @constructs
  7      * @extends BuildingBlockDescription
  8      */
  9     initialize: function($super, /** Hash */ properties) {
 10         $super(properties);
 11 
 12       
 13         this._screens = new Hash();
 14         this._preconditions = new Hash();
 15         this._postconditions = new Hash();
 16     },
 17 
 18     /**
 19      * to JSON object function
 20      * @type Object
 21      */
 22     toJSON: function() {
 23         var result = {
 24             "definition": {
 25                 "screens": this._getScreens(),
 26                 "preconditions": this.getPreconditions(),//.size() > 0 ? [this.getPreconditions()] : [],
 27                 "postconditions": this.getPostconditions()//.size() > 0 ? [this.getPostconditions()] : []
 28             }
 29         };
 30         result = Object.extend(result,{
 31             "name": this.name,
 32             "label": this.label,
 33             "tags": this.tags,
 34             "version": this.version,
 35             "id": this.id,
 36             "creator": this.creator,
 37             "description": this.description,
 38             "rights": this.rights,
 39             "creationDate": this.creationDate,
 40             "icon": this.icon,
 41             "screenshot": this.screenshot,
 42             "homepage": this.homepage,
 43             "type": "screenflow"
 44         });
 45         return result;
 46     },
 47     
 48     /**
 49      * Adds a new screen.
 50      * @param ScreenDescription
 51      *      Screen to be added to the
 52      *      Screenflow document.
 53      */
 54     addScreen: function (/** ScreenInstance */ instance, /** Object */ position) {
 55         this._screens.set(instance.getUri(), {
 56             "buildingblock":   instance,
 57             "position": position
 58         });
 59     },
 60 
 61     /**
 62      * Updates the position of the screen
 63      */
 64     updateScreen: function (/** String */ uri, /** Object */ position) {
 65         this._screens.get(uri).position = position;
 66     },
 67 
 68     /**
 69      * Delete a screen.
 70      * @param ScreenDescription
 71      *      Screen to be deleted from the
 72      *      Screenflow document.
 73      */
 74     remove: function(/** String */ uri) {
 75         this._screens.unset(uri);
 76         this._preconditions.unset(uri);
 77         this._postconditions.unset(uri);
 78     },
 79     /**
 80      * Adds a new *-condition to the screenflow description
 81      */
 82     addPrePost: function(/** PrePostInstance */ instance, /** Object */ position) {
 83         switch(instance.getType()) {
 84             case 'pre':
 85                 this._preconditions.set(instance.getUri(), {'buildingblock': instance,
 86                                                             'position': position});
 87                 break;
 88             case 'post':
 89                 this._postconditions.set(instance.getUri(), {'buildingblock': instance,
 90                                                             'position': position});
 91                 break;
 92             default:
 93                 //Do nothing
 94         }  
 95     },
 96 
 97     updatePrePost: function(/** String */ uri, /** Object */ position) {
 98         var condition = this._preconditions.get(uri);
 99         if (condition) {
100             condition.position = position;
101         } else {
102             condition = this._postconditions.get(uri);
103             condition.position = position;
104         }
105     },
106     
107     
108     /**
109      * Implementing the TableModel interface
110      * @type Array
111      */
112     getInfo: function() {
113         var info = new Hash();
114         info.set('Title', this.getTitle());
115         info.set('Tags', this.tags.collect(function(tag) {
116                 return tag.label['en-gb'];
117             }).join(", "));
118         info.set('Version', this.version);
119         return info;
120     },
121     
122     /**
123      * Get a list of preconditions in form of a JSON Object
124      * @type Array
125      */
126     getPreconditions: function() {
127         var list = new Array();
128         this._preconditions.values().each(function(pre) {
129             var element = Object.extend(pre.buildingblock.toJSONForScreenflow(), {'position':
130                 pre.position});            
131             list.push(element);
132         }.bind(this));
133         return list;
134     },
135 
136     /**
137      * Get a list of postconditions in form of a JSON Object
138      * @type Array
139      */
140     getPostconditions: function() {
141         var list = new Array();
142         this._postconditions.values().each(function(post) {
143             var element = Object.extend(post.buildingblock.toJSONForScreenflow(), {'position':
144                 post.position});
145             list.push(element);
146         }.bind(this));
147         return list;
148     },
149 
150     /**
151      * Get a list of canvas instances
152      * @type Array
153      */
154     getCanvasInstances: function() {
155         var result = new Array();
156         var list = this._screens.values().concat(this._preconditions.values());
157         list = list.concat(this._postconditions.values());
158         list.each(function(instance){
159             result.push(instance.buildingblock);
160         });
161         return result;
162     },
163 
164     /**
165      * Returns true if an element with the parameter uri is in the screen
166      * @type Boolean
167      */
168     contains: function(/** String */ uri) {
169         var list = this._screens.values().concat(this._preconditions.values());
170         list = list.concat(this._postconditions.values());
171         var element = list.detect(function(instance) {
172                     return instance.buildingblock.getUri() == uri;
173         });
174         if (element) {
175             return true;
176         } else {
177             return false;
178         }
179     },
180 
181     /**
182      * Necessary for compatibility with other BuildingblockDescriptions
183      * @type Array
184      */
185     getConditionInstances: function() {
186         return new Array();
187     },
188     
189     //************************ PRIVATE METHODS *******************//
190     /**
191      * @type Array
192      * @private
193      */
194     _getScreenUris: function() {
195         return this._screens.keys();
196     },
197 
198     /**
199      * Return the list of screens
200      * @type Array
201      * @private
202      */
203     _getScreens: function() {
204         var result = new Array();
205         this._screens.values().each(function(screen){
206             result.push({
207                 'uri': screen.buildingblock.getUri(),
208                 'position': screen.position
209             });
210         });
211         return result;
212     }
213 });
