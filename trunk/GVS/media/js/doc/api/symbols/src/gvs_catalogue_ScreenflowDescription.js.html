1 var ScreenflowDescription = Class.create(BuildingBlockDescription,
  2     /** @lends ScreenflowDescription.prototype */ {
  3 
  4     /**
  5      * Screenflow building block description.
  6      * @constructs
  7      * @extends BuildingBlockDescription
  8      */
  9     initialize: function($super, /** Hash */ properties) {
 10         this.definition = new Object();
 11         this.definition.screens = new Array();
 12         this.definition.preconditions = new Array();
 13         this.definition.postconditions = new Array();
 14         this.description = new Object();
 15         this.label = new Object();
 16         this.domainContext = new Array();
 17         
 18         $super(properties);
 19     },
 20     
 21     /**
 22      * Adds a new screen.
 23      * @param ScreenDescription
 24      *      Screen to be added to the
 25      *      Screenflow document.
 26      */
 27     addScreen: function (/** String */ uri, /** Object */ position) {
 28         this.definition.screens.push({
 29             "uri":   uri,
 30             "position": position
 31         });
 32     },
 33 
 34     /**
 35      * Delete a screen.
 36      * @param ScreenDescription
 37      *      Screen to be deleted from the
 38      *      Screenflow document.
 39      */
 40     removeScreen: function(/** String */ id) { 
 41         for (var i=0; i < this.definition.screens.length; i++) {
 42             if (this.definition.screens[i].uri == id) {
 43                 this.definition.screens.splice(i,1);
 44                 break;
 45             }
 46         }
 47     },
 48     /**
 49      * Adds a new *-condition to the screenflow description
 50      */
 51     addPrePost: function(/** PrePostInstance */ instance) {
 52         switch(instance.getType()) {
 53             case 'pre':
 54                 this.definition.preconditions.push(instance.getProperties());
 55                 break;
 56             case 'post':
 57                 this.definition.postconditions.push(instance.getProperties());
 58                 break;
 59             default:
 60                 //Do nothing
 61         }  
 62     },
 63     /**
 64      * Removing a *-condition
 65      */
 66     removePrePost: function(/** String */ id) {
 67         var found = false;
 68         for (var i=0; i < this.definition.preconditions.length; i++) {
 69             if (this.definition.preconditions[i].uri == id) {
 70                 this.definition.preconditions.splice(i,1);
 71                 found = true;
 72                 break;
 73             }
 74         }
 75         if (!found) {
 76             for (var i=0; i < this.definition.preconditions.length; i++) {
 77                 if (this.definition.preconditions[i].uri == id) {
 78                     this.definition.preconditions.splice(i,1);
 79                     found = true;
 80                     break;
 81                 }
 82             }            
 83         }       
 84     },
 85     
 86     /**
 87      * Implementing the TableModel interface
 88      * @type String
 89      */
 90     getTitle: function() {
 91         return this.label['en-gb'];    
 92     },
 93     
 94     /**
 95      * Implementing the TableModel interface
 96      * @type Array
 97      */
 98     getInfo: function() {
 99         var info = new Hash();
100         info.set('Title', this.getTitle());
101         info.set('Tags', this.domainContext.tags.join(", "));
102         info.set('Version', this.version);
103         return info;
104     },
105     
106     
107     
108     //************************ PRIVATE METHODS *******************//
109     /**
110      * @type Array
111      */
112     _getScreenUris: function() {
113         var uris = [];
114         this.screens.each(function (pair) {
115             uris.push(pair.key);
116         });
117         return uris;
118     }
119 });
