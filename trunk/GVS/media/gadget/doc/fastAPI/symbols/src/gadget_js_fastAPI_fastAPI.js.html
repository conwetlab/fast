 41 FastBaseAPI.Request = Class.create(/** @lends FastAPI.Request.prototype */{
 42     /**
 43      * Creates an object to make AJAX requests
 44      * @constructs
 45      * @param url
 46      *      url to be requested
 47      * @param options
 48      * <li>method: 'get' || 'post' || 'put' || 'delete'</li>
 49      * <li> content: 'xml' || 'text' || 'json'</li>
 50      * <li> context: context of the invoking object</li>
 51      * <li> parameters: either as a URL-encoded string or as any Hash-compatible object.</li>
 52      * <li> requestHeaders: a javascript object with properties representing headers.</li>
 53      * <li> onSuccess: Invoked when a request completes and its status code is undefined or belongs in the 2xy family.</li>
 54      * <li> onFailure: Invoked when a request completes and its status code exists but is not in the 2xy family.</li>
 55      * <li> onXYZ: (with XYZ representing any HTTP status code): Invoked just after the response</li>
 56      *       is completed if the status code matchs the code used in the callback name. Prevents the execution of
 57      *       onSuccess and onFailure. Happens before onComplete.</li>
 58      * <li> onException: Fires when there was an exception while dispatching other callbacks.</li>
 59      */
 60     initialize: function(url, options) {
 61         this.url = url;
 62 
 63         this.options = {
 64             method:         'post',
 65             content:        '',
 66             onSuccess:      Prototype.emptyFunction,
 67             onFailure:      Prototype.emptyFunction
 68         };
 69         Object.extend(this.options, options || { });
 70 
 71         this.options.method = this.options.method.toLowerCase();
 72 
 73         if (Object.isString(this.options.parameters))
 74             this.options.parameters = this.options.parameters.toQueryParams();
 75         else if (Object.isHash(this.options.parameters))
 76             this.options.parameters = this.options.parameters.toObject();
 77         this.setRequestHeaders();
 78         this.call();
 79     },
 80 
 81     /**
 82      * Sets the default headers.
 83      * This method can be overriden and augmented by the implementations.
 84      *  - Accept defaults to 'text/javascript, text/html, application/xml, text/xml, *\/*'
 85      *  - Content-type is built based on the contentType and encoding options.
 86      * @private
 87      */
 88     setRequestHeaders: function() {
 89         var headers = {
 90             'Accept': 'text/javascript, text/html, application/xml, text/xml, application/json, */*'
 91         };
 92 
 93         if (this.options.method == 'post') {
 94             headers['Content-type'] = this.options.contentType +
 95                 (this.options.encoding ? '; charset=' + this.options.encoding : '');
 96         }
 97 
 98         Object.extend(headers, this.options.requestHeaders || { });
 99         this.options.requestHeaders = headers;
100     },
101 
102     /**
103      * Make a general-purpose request to a remote server.
104      * @private
105      */
106     call: function() {
107         throw 'Abstract Method invocation. ' +
108               'FastAPI.Request :: request';
109     }
110 });
